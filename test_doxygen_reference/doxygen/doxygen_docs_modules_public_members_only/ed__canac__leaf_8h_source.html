200 // Convert and return the results

202 update(conductance_stomatal_h2o_op, second_c4photoC_output.Gs * 1e-3); // mol / m^2 / s
203 update(assimilation_gross_op, second_c4photoC_output.GrossAssim * 1e-6); // mol / m^2 / s

320Light_profile sunML(
321 double ambient_ppfd_beam, // micromol / (m^2 beam) / s
322 double ambient_ppfd_diffuse, // micromol / m^2 / s
323 double lai, // dimensionless from m^2 / m^2
324 int nlayers, // dimensionless
325 double cosine_zenith_angle, // dimensionless
326 double kd, // dimensionless
327 double chil, // dimensionless from m^2 / m^2
328 double absorptivity, // dimensionless from mol / mol
329 double heightf, // m^-1 from m^2 leaf / m^2 ground / m height
330 double par_energy_content, // J / micromol
331 double par_energy_fraction, // dimensionless
332 double leaf_transmittance, // dimensionless
333 double leaf_reflectance // dimensionless
334)
335{
336 if (nlayers < 1 || nlayers > MAXLAY) {
337 throw std::out_of_range("nlayers must be at least 1 but no more than MAXLAY.");
338 }
339 if (cosine_zenith_angle > 1 || cosine_zenith_angle < -1) {
340 throw std::out_of_range("cosine_zenith_angle must be between -1 and 1.");
341 }
342 if (kd > 1 || kd < 0) {
343 throw std::out_of_range("kd must be between 0 and 1.");
344 }
345 if (chil < 0) {
346 throw std::out_of_range("chil must be non-negative.");
347 }
348 if (absorptivity > 1 || absorptivity < 0) {
349 throw std::out_of_range("absorptivity must be between 0 and 1.");
350 }
351 if (heightf <= 0) {
352 throw std::out_of_range("heightf must greater than zero.");
353 }
354
355 // Calculate the leaf shape factor for an ellipsoidal leaf angle
356 // distribution using the equation from page 251 of Campbell & Norman
357 // (1998). We will use this value as `k`, the canopy extinction coefficient
358 // for photosynthetically active radiation throughout the canopy. This
359 // quantity represents the ratio of horizontal area to total area for leaves
360 // in the canopy and is therefore dimensionless from
361 // (m^2 ground) / (m^2 leaf).
362 double zenith_angle = acos(cosine_zenith_angle); // radians
363 double k0 = sqrt(pow(chil, 2) + pow(tan(zenith_angle), 2));
364 double k1 = chil + 1.744 * pow((chil + 1.182), -0.733);
365 double k = k0 / k1; // dimensionless
366
367 double lai_per_layer = lai / nlayers;
368
369 // Calculate the ambient direct PPFD through a surface parallel to the ground
370 const double ambient_ppfd_beam_ground = ambient_ppfd_beam * cosine_zenith_angle; // micromol / (m^2 ground) / s
371
372 // Calculate the ambient direct PPFD through a unit area of leaf surface
373 double ambient_ppfd_beam_leaf = ambient_ppfd_beam_ground * k; // micromol / (m^2 leaf) / s
374
375 Light_profile light_profile;
376 for (int i = 0; i < nlayers; ++i) {
377 // Get the cumulative LAI for this layer, which represents the total
378 // leaf area above this layer
379 const double cumulative_lai = lai_per_layer * (i + 0.5);
380
381 // Calculate the amount of PPFD scattered out of the direct beam using
382 // Equations 15.6 and 15.1 from Campbell & Norman (1998), following
383 // example 15.2. This is a diffuse flux density representing the flux
384 // through any surface.
385 const double scattered_ppfd =
386 ambient_ppfd_beam_ground * (exp(-k * sqrt(absorptivity) * cumulative_lai) -
387 exp(-k * cumulative_lai)); // micromol / m^2 / s
388
389 // Calculate the total flux of diffuse photosynthetically active light
390 // in this layer by combining the scattered PPFD with the ambient
391 // diffuse PPFD. Here we use Equation 15.6 with `alpha` = 1 and
392 // `kbe(phi)` = kd.
393 double diffuse_ppfd =
394 ambient_ppfd_diffuse * exp(-kd * cumulative_lai) + scattered_ppfd; // micromol / m^2 / s
395
396 // Calculate the fraction of sunlit and shaded leaves in this canopy
397 // layer using Equation 15.21.
398 const double Ls = (1 - exp(-k * lai_per_layer)) * exp(-k * cumulative_lai) / k; // dimensionless
399 double sunlit_fraction = Ls / lai_per_layer; // dimensionless
400 double shaded_fraction = 1 - sunlit_fraction; // dimensionless
401
402 // Calculate an "average" incident PPFD for the sunlit and shaded leaves
403 // that doesn't seem to be based on a formula from Campbell & Norman
404 // (1998). It's interpreted as a flux density through a unit of leaf
405 // area, but that may not be correct.
406 double average_ppfd =
407 (sunlit_fraction * (ambient_ppfd_beam_leaf + diffuse_ppfd) + shaded_fraction * diffuse_ppfd) *
408 (1 - exp(-k * lai_per_layer)) / k; // micromol / (m^2 leaf) / s
409
410 // For values of cosine_zenith_angle close to or less than 0, in place
411 // of the calculations above, we want to use the limits of the above
412 // expressions as cosine_zenith_angle approaches 0 from the right:
413 if (cosine_zenith_angle <= 1E-10) {
414 ambient_ppfd_beam_leaf = ambient_ppfd_beam / k1;
415 diffuse_ppfd = ambient_ppfd_diffuse * exp(-kd * cumulative_lai);
416 sunlit_fraction = 0;
417 shaded_fraction = 1;
418 average_ppfd = 0;
419 }
420
421 // Store values of incident PPFD
422 light_profile.sunlit_incident_ppfd[i] = ambient_ppfd_beam_leaf + diffuse_ppfd; // micromol / (m^2 leaf) / s
423 light_profile.incident_ppfd_scattered[i] = scattered_ppfd; // micromol / m^2 / s
424 light_profile.shaded_incident_ppfd[i] = diffuse_ppfd; // micromol / (m^2 leaf) / s
425 light_profile.average_incident_ppfd[i] = average_ppfd; // micromol / (m^2 leaf) / s
426 light_profile.sunlit_fraction[i] = sunlit_fraction; // dimensionless from m^2 / m^2
427 light_profile.shaded_fraction[i] = shaded_fraction; // dimensionless from m^2 / m^2
428 light_profile.height[i] = (lai - cumulative_lai) / heightf; // m
429
430 // Store values of absorbed PPFD
431 light_profile.sunlit_absorbed_ppfd[i] =
432 thin_layer_absorption(
433 leaf_reflectance,
434 leaf_transmittance,
435 ambient_ppfd_beam_leaf + diffuse_ppfd); // micromol / m^2 / s
436
437 light_profile.shaded_absorbed_ppfd[i] =
438 thin_layer_absorption(
439 leaf_reflectance,
440 leaf_transmittance,
441 diffuse_ppfd); // micromol / m^2 / s
442
443 // Store values of absorbed solar energy (including PAR and NIR)
444 light_profile.sunlit_absorbed_shortwave[i] =
445 absorbed_shortwave_from_incident_ppfd(
446 ambient_ppfd_beam_leaf + diffuse_ppfd,
447 par_energy_content,
448 par_energy_fraction,
449 leaf_reflectance,
450 leaf_transmittance); // J / (m^2 leaf) / s
451
452 light_profile.shaded_absorbed_shortwave[i] =
453 absorbed_shortwave_from_incident_ppfd(
454 diffuse_ppfd,
455 par_energy_content,
456 par_energy_fraction,
457 leaf_reflectance,
458 leaf_transmittance); // J / (m^2 leaf) / s
459
460 light_profile.average_absorbed_shortwave[i] =
461 absorbed_shortwave_from_incident_ppfd(
462 average_ppfd,
463 par_energy_content,
464 par_energy_fraction,
465 leaf_reflectance,
466 leaf_transmittance); // J / (m^2 leaf) / s
467 }
468 return light_profile;

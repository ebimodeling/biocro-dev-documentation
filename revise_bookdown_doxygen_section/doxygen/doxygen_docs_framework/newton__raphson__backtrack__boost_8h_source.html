75 // bounds, rescale it. Here we assume that x_old lies within the bounds. We
76 // check each element in the following manner:
77 //
78 // If an element lies above the upper bound, find an appropriate rescaling
79 // factor r such that x_old[i] + direction[i] * r = upper_bounds[i],
80 // i.e., r = (upper_bounds[i] - x_old[i]) / direction[i].
81 //
82 // If an element lies below the lower bound, find an appropriate rescaling
83 // factor r such that x_old[i] + direction[i] * r = lower_bounds[i],
84 // i.e., r = (lower_bounds[i] - x_old[i]) / direction[i].

275 // so this is the first backtrack.

277 temporary_lambda = -slope / (2.0 * (f_scalar_new - f_scalar_old - slope));
278 message += " doing first backtrack\n";

293 temporary_lambda = -slope / (2.0 * b);
294 } else {

296 // value of lambda given by Equation 9.7.14.
297 // Begin solving by calculating the value under the square root,
298 // called `disc`

300 if (disc < 0.0) {

302 // when the cubic approximation to g has no minimum.
303 temporary_lambda = 0.5 * lambda;
304 } else if (b <= 0.0) {
305 // Use Equation 9.7.14
306 temporary_lambda = (-b + sqrt(disc)) / (3.0 * a);
307 } else {
308 // Not sure why we do this, but it's part of the code
309 // in Numerical Recipes in C.
310 temporary_lambda = -slope / (b + sqrt(disc));
311 }
312 }
313
314 // Ensure that the new lambda value is smaller than half of the previous value
315 temporary_lambda = std::min(temporary_lambda, 0.5 * lambda);
316 }
317
318 // Store some values for later loops
319 lambda_2 = lambda;
320 f_scalar_2 = f_scalar_new;
321
322 // Ensure that the new lambda value is larger than 10% of the previous value
323 lambda = std::max(temporary_lambda, 0.1 * lambda);
324
325 if (nrb_print) {
326 Rprintf(message.c_str()); // print the message now in case an error is thrown
327 }
328 }
329 } while (found_acceptable_step == false && found_possible_local_min == false);

334 Rprintf(message.c_str());
335 }

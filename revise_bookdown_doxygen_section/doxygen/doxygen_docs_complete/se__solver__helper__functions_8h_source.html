38template <typename equation_ptr_type, typename vector_type>
39bool adjust_bad_guess_random(
40 equation_ptr_type const& se,
41 vector_type const& lower_bounds,
42 vector_type const& upper_bounds,
43 vector_type& bad_guess,
44 vector_type& difference_vector_at_bad_guess)
45{
46 // Determine which elements lie outside the bounds
47 std::vector<bool> oob = is_outside_bounds(bad_guess, lower_bounds, upper_bounds);
48
49 // Adjust the problematic elements
50 int num_adjusted = 0;
51 for (size_t i = 0; i < bad_guess.size(); ++i) {
52 if (oob[i]) {
53 bad_guess[i] = lower_bounds[i] + se_solver_rand::rand_01() * (upper_bounds[i] - lower_bounds[i]);
54 ++num_adjusted;
55 }
56 }
57
58 // Evaluate the difference vector at the new guess if an adjustment was made
59 if (num_adjusted > 0) {
60 (*se)(bad_guess, difference_vector_at_bad_guess); // modifies difference_vector_at_bad_guess
61 }
62
63 return num_adjusted > 0;
64}

189template <typename vector_type>
190std::vector<bool> has_not_converged_rel(
191 vector_type const& difference_vector,
192 vector_type const& guess,
193 vector_type const& tolerances)
194{
195 std::vector<bool> result(difference_vector.size());
196
197 std::string message = "Checking relative convergence:\n";
198 char buff[128];
199
200 for (size_t i = 0; i < difference_vector.size(); ++i) {
201 double threshold = fabs(guess[i]) * tolerances[i];
202 if (threshold < calculation_constants::eps_zero) {
203 // If guess[i] is zero, a relative comparison doesn't make sense.
204 // Treat `tolerance` as an absolute threshold in this case.
205 threshold = tolerances[i];
206 }
207 result[i] = fabs(difference_vector[i]) > threshold;
208 sprintf(buff, " difference = %e, guess = %e, tolerance = %e, threshold = %e, result = %i\n", difference_vector[i], guess[i], tolerances[i], threshold, (int)result[i]);
209 message += std::string(buff);
210 }
211
212 message += "\n";

235template <typename vector_type>
236double f_scalar_norm_from_f_vec(vector_type F_vec, vector_type F_vec_norm)
237{
238 double f_scalar_norm = 0.0;
239 for (size_t i = 0; i < F_vec.size(); ++i) {
240 if (F_vec_norm[i] < calculation_constants::eps_zero) {
241 f_scalar_norm += F_vec[i] * F_vec[i];
242 } else {
243 f_scalar_norm += F_vec[i] * F_vec[i] / (F_vec_norm[i] * F_vec_norm[i]);
244 }
245 }
246 return f_scalar_norm;
247}

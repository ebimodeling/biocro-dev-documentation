14double poisson_density(int x, double lambda)
15{
16 // The probability density for the Poisson distribution is
17 // e^-lambda * lambda^x / x!
18 // The factorial term produces numbers too large to hold, so perform the calculation in the log domain.
19 // n! can be estimated by the approximation sqrt(2 * pi * x) * (x / e)^x.
20 using math_constants::pi;
21 using math_constants::e;
22
23 // Stirling's approximation for n!:
24 double factorial_x = sqrt(2 * pi * x) * pow((x / e), x);
25
26 double log_result = -lambda + x * log(lambda) - log(factorial_x);
27
28 return exp(log_result);
29}

51Light_model lightME(double cosine_zenith_angle, double atmospheric_pressure)
52{
53 double direct_irradiance_transmittance;
54 double diffuse_irradiance_transmittance;
55
56 if (cosine_zenith_angle <= 0) { // Check if the Sun is at or below the horizon. If it is, directly set direct_irradiance_transmittance and diffuse_irradiance_transmittance because the equations below would otherwise produce erroneous results. The Sun is at or below the horizon when cosine_zenith_angle is less than or equal to 0.
57 direct_irradiance_transmittance = 0;
58 diffuse_irradiance_transmittance = 1;
59 } else { // If the Sun is above the horizon, calculate diffuse and direct irradiance from the Sun's angle to the ground and the path through the atmosphere.
60 constexpr double atmospheric_transmittance = 0.85; // dimensionless.
61 const double pressure_ratio = atmospheric_pressure / physical_constants::atmospheric_pressure_at_sea_level; // dimensionless.
62 constexpr double proportion_of_irradiance_scattered = 0.3; // dimensionless.
63
64 direct_irradiance_transmittance = pow(atmospheric_transmittance, (pressure_ratio / cosine_zenith_angle)); // Modified from equation 11.11 of Norman and Campbell.
65 diffuse_irradiance_transmittance = proportion_of_irradiance_scattered * (1 - direct_irradiance_transmittance) * cosine_zenith_angle; // Modified from equation 11.13 of Norman and Campbell.
66 }
67
68 Light_model light_model;
69 light_model.direct_irradiance_fraction = direct_irradiance_transmittance / (direct_irradiance_transmittance + diffuse_irradiance_transmittance); // dimensionless.
70 light_model.diffuse_irradiance_fraction = diffuse_irradiance_transmittance / (direct_irradiance_transmittance + diffuse_irradiance_transmittance); // dimensionless.
71
72 return light_model;
73}

109double absorbed_shortwave_from_incident_ppfd(
110 double incident_ppfd, // micromol / m^2 / s
111 double par_energy_content, // J / micromol
112 double par_energy_fraction, // dimensionless
113 double leaf_reflectance, // dimensionless
114 double leaf_transmittance // dimensionless
115)
116{
117 double incident_par = incident_ppfd * par_energy_content; // J / m^2 / s
118
119 double incident_nir = incident_par *
120 (1 - par_energy_fraction) /
121 par_energy_fraction; // J / m^2 /s
122
123 double incident_shortwave = incident_par + incident_nir; // J / m^2 / s
124
125 return incident_shortwave *
126 (1 - leaf_reflectance - leaf_transmittance) /
127 (1 - leaf_transmittance); // J / m^2 / s
128}

174Light_profile sunML(
175 double ambient_ppfd_beam, // micromol / (m^2 beam) / s
176 double ambient_ppfd_diffuse, // micromol / m^2 / s
177 double lai, // dimensionless from m^2 / m^2
178 int nlayers, // dimensionless
179 double cosine_zenith_angle, // dimensionless
180 double kd, // dimensionless
181 double chil, // dimensionless from m^2 / m^2
182 double absorptivity, // dimensionless from mol / mol
183 double heightf, // m^-1 from m^2 leaf / m^2 ground / m height
184 double par_energy_content, // J / micromol
185 double par_energy_fraction, // dimensionless
186 double leaf_transmittance, // dimensionless
187 double leaf_reflectance // dimensionless
188)
189{
190 if (nlayers < 1 || nlayers > MAXLAY) {
191 throw std::out_of_range("nlayers must be at least 1 but no more than MAXLAY.");
192 }
193 if (cosine_zenith_angle > 1 || cosine_zenith_angle < -1) {
194 throw std::out_of_range("cosine_zenith_angle must be between -1 and 1.");
195 }
196 if (kd > 1 || kd < 0) {
197 throw std::out_of_range("kd must be between 0 and 1.");
198 }
199 if (chil < 0) {
200 throw std::out_of_range("chil must be non-negative.");
201 }
202 if (absorptivity > 1 || absorptivity < 0) {
203 throw std::out_of_range("absorptivity must be between 0 and 1.");
204 }
205 if (heightf <= 0) {
206 throw std::out_of_range("heightf must greater than zero.");
207 }
208
209 // Calculate the leaf shape factor for an ellipsoidal leaf angle
210 // distribution using the equation from page 251 of Campbell & Norman
211 // (1998). We will use this value as `k`, the canopy extinction coefficient
212 // for photosynthetically active radiation throughout the canopy. This
213 // quantity represents the ratio of horizontal area to total area for leaves
214 // in the canopy and is therefore dimensionless from
215 // (m^2 ground) / (m^2 leaf).
216 double zenith_angle = acos(cosine_zenith_angle); // radians
217 double k0 = sqrt(pow(chil, 2) + pow(tan(zenith_angle), 2));
218 double k1 = chil + 1.744 * pow((chil + 1.182), -0.733);
219 double k = k0 / k1; // dimensionless
220
221 double lai_per_layer = lai / nlayers;
222
223 // Calculate the ambient direct PPFD through a surface parallel to the ground
224 const double ambient_ppfd_beam_ground = ambient_ppfd_beam * cosine_zenith_angle; // micromol / (m^2 ground) / s
225
226 // Calculate the ambient direct PPFD through a unit area of leaf surface
227 double ambient_ppfd_beam_leaf = ambient_ppfd_beam_ground * k; // micromol / (m^2 leaf) / s
228
229 Light_profile light_profile;
230 for (int i = 0; i < nlayers; ++i) {
231 // Get the cumulative LAI for this layer, which represents the total
232 // leaf area above this layer
233 const double cumulative_lai = lai_per_layer * (i + 0.5);
234
235 // Calculate the amount of PPFD scattered out of the direct beam using
236 // Equations 15.6 and 15.1 from Campbell & Norman (1998), following
237 // example 15.2. This is a diffuse flux density representing the flux
238 // through any surface.
239 const double scattered_ppfd =
240 ambient_ppfd_beam_ground * (exp(-k * sqrt(absorptivity) * cumulative_lai) -
241 exp(-k * cumulative_lai)); // micromol / m^2 / s
242
243 // Calculate the total flux of diffuse photosynthetically active light
244 // in this layer by combining the scattered PPFD with the ambient
245 // diffuse PPFD. Here we use Equation 15.6 with `alpha` = 1 and
246 // `kbe(phi)` = kd.
247 double diffuse_ppfd =
248 ambient_ppfd_diffuse * exp(-kd * cumulative_lai) + scattered_ppfd; // micromol / m^2 / s
249
250 // Calculate the fraction of sunlit and shaded leaves in this canopy
251 // layer using Equation 15.21.
252 const double Ls = (1 - exp(-k * lai_per_layer)) * exp(-k * cumulative_lai) / k; // dimensionless
253 double sunlit_fraction = Ls / lai_per_layer; // dimensionless
254 double shaded_fraction = 1 - sunlit_fraction; // dimensionless
255
256 // Calculate an "average" incident PPFD for the sunlit and shaded leaves
257 // that doesn't seem to be based on a formula from Campbell & Norman
258 // (1998). It's interpreted as a flux density through a unit of leaf
259 // area, but that may not be correct.
260 double average_ppfd =
261 (sunlit_fraction * (ambient_ppfd_beam_leaf + diffuse_ppfd) + shaded_fraction * diffuse_ppfd) *
262 (1 - exp(-k * lai_per_layer)) / k; // micromol / (m^2 leaf) / s
263
264 // For values of cosine_zenith_angle close to or less than 0, in place
265 // of the calculations above, we want to use the limits of the above
266 // expressions as cosine_zenith_angle approaches 0 from the right:
267 if (cosine_zenith_angle <= 1E-10) {
268 ambient_ppfd_beam_leaf = ambient_ppfd_beam / k1;
269 diffuse_ppfd = ambient_ppfd_diffuse * exp(-kd * cumulative_lai);
270 sunlit_fraction = 0;
271 shaded_fraction = 1;
272 average_ppfd = 0;
273 }
274
275 // Store these values of incident PPFD
276 light_profile.sunlit_incident_ppfd[i] = ambient_ppfd_beam_leaf + diffuse_ppfd; // micromole / (m^2 leaf) / s
277 light_profile.incident_ppfd_scattered[i] = scattered_ppfd; // micromole / m^2 / s
278 light_profile.shaded_incident_ppfd[i] = diffuse_ppfd; // micromole / (m^2 leaf) / s
279 light_profile.average_incident_ppfd[i] = average_ppfd; // micromole / (m^2 leaf) / s
280 light_profile.sunlit_fraction[i] = sunlit_fraction; // dimensionless from m^2 / m^2
281 light_profile.shaded_fraction[i] = shaded_fraction; // dimensionless from m^2 / m^2
282 light_profile.height[i] = (lai - cumulative_lai) / heightf; // m
283
284 // We also need to determine the total amount of absorbed solar energy
285 // for sunlit and shaded leaves
286 light_profile.sunlit_absorbed_shortwave[i] =
287 absorbed_shortwave_from_incident_ppfd(
288 ambient_ppfd_beam_leaf + diffuse_ppfd,
289 par_energy_content,
290 par_energy_fraction,
291 leaf_reflectance,
292 leaf_transmittance); // J / (m^2 leaf) / s
293
294 light_profile.shaded_absorbed_shortwave[i] =
295 absorbed_shortwave_from_incident_ppfd(
296 diffuse_ppfd,
297 par_energy_content,
298 par_energy_fraction,
299 leaf_reflectance,
300 leaf_transmittance); // J / (m^2 leaf) / s
301
302 light_profile.average_absorbed_shortwave[i] =
303 absorbed_shortwave_from_incident_ppfd(
304 average_ppfd,
305 par_energy_content,
306 par_energy_fraction,
307 leaf_reflectance,
308 leaf_transmittance); // J / (m^2 leaf) / s
309 }
310 return light_profile;
311}

629struct ET_Str EvapoTrans2(
630 double absorbed_shortwave_radiation_et, // J / m^2 / s (used to calculate evapotranspiration rate)
631 double absorbed_shortwave_radiation_lt, // J / m^2 / s (used to calculate leaf temperature)
632 double airTemp, // degrees C
633 double RH, // dimensionless from Pa / Pa
634 double WindSpeed, // m / s
635 double stomatal_conductance, // mmol / m^2 / s
636 double leaf_width, // meter
637 double specific_heat_of_air, // J / kg / K
638 double minimum_gbw, // mol / m^2 / s
639 int eteq // unitless parameter
640)
641{
642 const double DdryA = TempToDdryA(airTemp); // kg / m^3. Density of dry air.,
643 const double LHV = TempToLHV(airTemp); // J / kg
644 const double SlopeFS = TempToSFS(airTemp); // kg / m^3 / K
645 const double SWVP = saturation_vapor_pressure(airTemp); // Pa.
646
647 // TODO: This is for about 20 degrees C at 100000 Pa. Change it to use the
648 // model state. (1 * R * temperature) / pressure
649 double constexpr volume_of_one_mole_of_air = 24.39e-3; // m^3 / mol
650
651 double minimum_gbw_in_m_per_s = minimum_gbw * volume_of_one_mole_of_air; // m / s
652
653 if (stomatal_conductance <= 0) {
654 throw std::range_error("Thrown in EvapoTrans2: stomatal conductance is not positive.");
655 }
656
657 double conductance_in_m_per_s = stomatal_conductance * 1e-3 * volume_of_one_mole_of_air; // m / s
658
659 if (RH > 1) {
660 throw std::range_error("Thrown in EvapoTrans2: RH (relative humidity) is greater than 1.");
661 }
662
663 // Convert from vapor pressure to vapor density using the ideal gas law.
664 // This is approximately right for temperatures what won't kill plants.
665 const double SWVC =
666 SWVP / physical_constants::ideal_gas_constant /
667 (airTemp + conversion_constants::celsius_to_kelvin) * physical_constants::molar_mass_of_water; // kg / m^3
668
669 if (SWVC < 0) {
670 throw std::range_error("Thrown in EvapoTrans2: SWVC is less than 0.");
671 }
672
673 const double PsycParam = DdryA * specific_heat_of_air / LHV; // kg / m^3 / K
674
675 const double vapor_density_deficit = SWVC * (1 - RH); // kg / m^3
676
677 const double ActualVaporPressure = RH * SWVP; // Pa
678
679 /* This is the original from WIMOVAC*/
680 double Deltat = 0.01; // degrees C
681 double ga;
682 double rlc; /* Long wave radiation for iterative calculation */
683 {
684 double ChangeInLeafTemp = 10.0; // degrees C
685 double Counter = 0;
686 do {
687 ga = leaf_boundary_layer_conductance_nikolov(
688 WindSpeed, leaf_width, airTemp, Deltat, conductance_in_m_per_s,
689 ActualVaporPressure, minimum_gbw_in_m_per_s); // m / s
690
691 /* In WIMOVAC, ga was added to the canopy conductance */
692 /* ga = (ga * gbcW)/(ga + gbcW); */
693
694 double OldDeltaT = Deltat;
695
696 rlc = 4 * physical_constants::stefan_boltzmann * pow(conversion_constants::celsius_to_kelvin + airTemp, 3) * Deltat; // W / m^2
697
698 /* rlc = net long wave radiation emittted per second
699 * = radiation emitted per second - radiation absorbed per second
700 * = sigma * (Tair + deltaT)^4 - sigma * Tair^4
701 *
702 * To make it a linear function of deltaT, do a Taylor series about
703 * deltaT = 0 and keep only the zero and first order terms.
704 *
705 * rlc = sigma * Tair^4 + deltaT * (4 * sigma * Tair^3) - sigma * Tair^4
706 * = 4 * sigma * Tair^3 * deltaT
707 *
708 * where 4 * sigma * Tair^3 is the derivative of
709 * sigma * (Tair + deltaT)^4 evaluated at deltaT = 0
710 */
711
712 const double PhiN2 = absorbed_shortwave_radiation_lt - rlc; // W / m^2
713
714 /* This equation is from Thornley and Johnson pg. 418 */
715 const double TopValue = PhiN2 * (1 / ga + 1 / conductance_in_m_per_s) - LHV * vapor_density_deficit; // J / m^3
716 const double BottomValue = LHV * (SlopeFS + PsycParam * (1 + ga / conductance_in_m_per_s)); // J / m^3 / K
717 Deltat = fmin(fmax(TopValue / BottomValue, -10), 10); // kelvin. Confine Deltat to the interval [-10, 10]:
718
719 ChangeInLeafTemp = fabs(OldDeltaT - Deltat); // kelvin
720 } while ((++Counter <= 10) && (ChangeInLeafTemp > 0.5));
721 }
722
723 /* Net radiation */
724 const double PhiN = fmax(0, absorbed_shortwave_radiation_et - rlc); // W / m^2
725
726 const double penman_monteith =
727 (SlopeFS * PhiN + LHV * PsycParam * ga * vapor_density_deficit) /
728 (LHV * (SlopeFS + PsycParam * (1 + ga / conductance_in_m_per_s))); // kg / m^2 / s. Thornley and Johnson. 1990. Plant and Crop Modeling. Equation 14.4k. Page 408.
729
730 const double EPen =
731 (SlopeFS * PhiN + LHV * PsycParam * ga * vapor_density_deficit) /
732 (LHV * (SlopeFS + PsycParam)); // kg / m^2 / s
733
734 const double EPries = 1.26 * SlopeFS * PhiN / (LHV * (SlopeFS + PsycParam)); // kg / m^2 / s
735
736 /* Choose equation to report */
737 double TransR;
738 switch (eteq) {
739 case 1:
740 TransR = EPen;
741 break;
742 case 2:
743 TransR = EPries;
744 break;
745 default:
746 TransR = penman_monteith;
747 break;
748 }
749
750 // TransR has units of kg / m^2 / s. Convert to mmol / m^2 / s using the
751 // molar mass of water (in kg / mol) and noting that 1e3 mmol = 1 mol
752 double cf = 1e3 / physical_constants::molar_mass_of_water; // mmol / kg for water
753
754 struct ET_Str et_results;
755 et_results.TransR = TransR * cf; // mmol / m^2 / s
756 et_results.EPenman = EPen * cf; // mmol / m^2 / s
757 et_results.EPriestly = EPries * cf; // mmol / m^2 / s
758 et_results.Deltat = Deltat; // degrees C
759 et_results.boundary_layer_conductance = ga / volume_of_one_mole_of_air; // mol / m^2 / s
760
761 return et_results;
762}

813double leaf_boundary_layer_conductance_nikolov(
814 double windspeed, // m / s
815 double leafwidth, // m
816 double air_temperature, // degrees C
817 double delta_t, // degrees C
818 double stomcond, // m / s
819 double water_vapor_pressure, // Pa
820 double minimum_gbw // m / s
821)
822{
823 constexpr double p = physical_constants::atmospheric_pressure_at_sea_level; // Pa
824
825 double leaftemp = air_temperature + delta_t; // degrees C
826 double gsv = stomcond; // m / s
827 double Tak = air_temperature + conversion_constants::celsius_to_kelvin; // K
828 double Tlk = leaftemp + conversion_constants::celsius_to_kelvin; // K
829 double ea = water_vapor_pressure; // Pa
830 double lw = leafwidth; // m
831
832 double esTl = saturation_vapor_pressure(leaftemp); // Pa.
833
834 // Forced convection
835 constexpr double cf = 1.6361e-3; // TODO: Nikolov et. al equation 29 use cf = 4.322e-3, not cf = 1.6e-3 as is used here.
836
837 double gbv_forced = cf * pow(Tak, 0.56) * pow((Tak + 120) * ((windspeed / lw) / p), 0.5); // m / s.
838
839 // Free convection
840 double gbv_free = gbv_forced;
841 double eb = (gsv * esTl + gbv_free * ea) / (gsv + gbv_free); // Pa. Eq 35
842
843 double Tvdiff = (Tlk / (1 - 0.378 * eb / p)) - (Tak / (1 - 0.378 * ea / p)); // kelvin. It is also degrees C since it is a temperature difference. Eq. 34
844
845 if (Tvdiff < 0) Tvdiff = -Tvdiff;
846
847 gbv_free = cf * pow(Tlk, 0.56) * pow((Tlk + 120) / p, 0.5) * pow(Tvdiff / lw, 0.25); // m / s. Eq. 33
848
849 // Overall conductance
850 double gbv = std::max(gbv_forced, gbv_free); // m / s
851
852 // Apply the minimum
853 return std::max(gbv, minimum_gbw); // m / s
854}

906double leaf_boundary_layer_conductance_thornley(
907 double CanopyHeight, // m
908 double WindSpeed, // m / s
909 double minimum_gbw, // m / s
910 double WindSpeedHeight // m
911)
912{
913 // Define constants used in the model
914 constexpr double kappa = 0.41; // dimensionless. von Karmon's constant. Thornley and Johnson pgs 414 and 416.
915 constexpr double ZetaCoef = 0.026; // dimensionless, Thornley and Johnson 1990, Eq. 14.9o
916 constexpr double ZetaMCoef = 0.13; // dimensionless, Thornley and Johnson 1990, Eq. 14.9o
917 constexpr double dCoef = 0.77; // dimensionless, Thornley and Johnson 1990, Eq. 14.9o.
918 // In the original text this value is reported as 0.64.
919 // In the 2000 reprinting of this text, the authors state that this value should be 0.77.
920 // See "Errata to the 2000 printing" on the page after the preface of the 2000 reprinting of the 1990 text.
921
922 // Apply the height limit
923 CanopyHeight = std::min(CanopyHeight, 0.98 * WindSpeedHeight); // meters
924
925 // Calculate terms that depend on the canopy height
926 const double Zeta = ZetaCoef * CanopyHeight; // meters
927 const double Zetam = ZetaMCoef * CanopyHeight; // meters
928 const double d = dCoef * CanopyHeight; // meters
929
930 // Calculate the boundary layer conductance `ga` according to Thornley and
931 // Johnson Eq. 14.9n, pg. 416
932 const double ga0 = pow(kappa, 2) * WindSpeed; // m / s
933 const double ga1 = log((WindSpeedHeight + Zeta - d) / Zeta); // dimensionless
934 const double ga2 = log((WindSpeedHeight + Zetam - d) / Zetam); // dimensionless
935 const double gbv = ga0 / (ga1 * ga2); // m / s
936
937 // Apply the minimum
938 return std::max(gbv, minimum_gbw); // m / s
939}

941/* Soil Evaporation Function */
942/* Variables I need */
943/* LAI = Leaf Area Index */
944/* k = extinction coefficient */
945/* ppfd = Photosynthetic photon flux density */
946/* awc, wiltp, fieldc = available water content, wilting point and field capacity */
947/* winds = wind speed */
948
949double SoilEvapo(
950 double LAI, double k, double air_temperature, double ppfd,
951 double soil_water_content, double fieldc, double wiltp, double winds,
952 double RelH, double rsec, double soil_clod_size, double soil_reflectance,
953 double soil_transmission, double specific_heat_of_air,
954 double par_energy_content)
955{
956 int method = 1;
957 /* A simple way of calculating the proportion of the soil that is hit by direct radiation. */
958 double soil_area_sunlit_fraction = exp(-k * LAI); // dimensionless.
959
960 /* For now the temperature of the soil will be the same as the air.
961 At a later time this can be made more accurate. I looked at the equations for this and the issue is that it is strongly dependent on
962 depth. Since the soil model now has a single layer, this cannot be implemented correctly at the moment. */
963 double SoilTemp = air_temperature;
964
965 /* From Campbell and Norman. Environmental Biophysics. */
966 /* If relative available water content is */
967 double rawc = (soil_water_content - wiltp) / (fieldc - wiltp); // dimensionless. relative available water content.
968
969 /* Campbell and Norman. Environmental Physics, page 142 */
970 double maximum_uptake_rate = 1 - pow((1 + 1.3 * rawc), -5); // dimensionless
971 /* This is a useful idea because dry soils evaporate little water when dry*/
972
973 /* Total Radiation */
974 /* At the moment soil evaporation is grossly overestimated. In WIMOVAC the light reaching the last layer of leaves is used. Here instead
975 of calculating this again, I will for now assume a 10% as a rough estimate. Note that I could maybe get this since layIdir and
976 layIDiff in sunML are external variables. Rprintf("ppfd %.5f",layIdir[0],"\n"); Update: 03-13-2009. I tried printing this
977 value but it is still too high and will likely overestimate soil evaporation. However, this is still a work in progress.
978 */
979 ppfd *= rsec; /* Radiation soil evaporation coefficient */
980
981 double TotalRadiation = ppfd * par_energy_content;
982
983 double DdryA = TempToDdryA(air_temperature);
984 double LHV = TempToLHV(air_temperature); // J / kg
985 double SlopeFS = TempToSFS(air_temperature);
986 double SWVC = saturation_vapor_pressure(air_temperature) * 1e-5;
987
988 double PsycParam = (DdryA * specific_heat_of_air) / LHV;
989 double vapor_density_deficit = SWVC * (1 - RelH / 100);
990
991 double BoundaryLayerThickness = 4e-3 * sqrt(soil_clod_size / winds);
992 double DiffCoef = 2.126e-5 + 1.48e-7 * SoilTemp;
993 double SoilBoundaryLayer = DiffCoef / BoundaryLayerThickness;
994
995 double Ja = 2 * TotalRadiation * ((1 - soil_reflectance - soil_transmission) / (1 - soil_transmission));
996
997 double rlc = 4 * physical_constants::stefan_boltzmann * pow((conversion_constants::celsius_to_kelvin + SoilTemp), 3) * 0.005;
998 /* the last term should be the difference between air temperature and soil. This is not actually calculated at the moment. Since this is
999 mostly relevant to the first soil layer where the temperatures are similar. I will leave it like this for now. */
1000
1001 double PhiN = Ja - rlc; /* Calculate the net radiation balance*/
1002 if (PhiN < 0) PhiN = 1e-7;
1003
1004 double Evaporation = 0.0;
1005 if (method == 0) {
1006 /* Priestly-Taylor */
1007 Evaporation = 1.26 * (SlopeFS * PhiN) / (LHV * (SlopeFS + PsycParam)); // kg / m^2 / s.
1008 } else {
1009 /* Penman-Monteith */
1010 Evaporation = (SlopeFS * PhiN + LHV * PsycParam * SoilBoundaryLayer * vapor_density_deficit) / (LHV * (SlopeFS + PsycParam)); // kg / m^2 / s.
1011 }
1012
1013 Evaporation *= soil_area_sunlit_fraction * maximum_uptake_rate; // kg / m^2 / s.
1014 if (Evaporation < 0) Evaporation = 1e-6; // Prevent odd values at very low light levels.
1015
1016 return Evaporation; // kg / m^2 / s.
1017}

1019double compute_wsPhoto(int wsFun, double fieldc, double wiltp, double phi1, double awc) {
1020 // Three different type of equations for modeling the effect of
1021 // water stress on vmax and leaf area expansion. The equation for
1022 // leaf area expansion is more severe than the one for vmax. */
1023 double wsPhoto;
1024 switch (wsFun) {
1025 case 0: { /* linear */
1026 double slp = 1 / (fieldc - wiltp);
1027 double intcpt = 1 - fieldc * slp;
1028 wsPhoto = slp * awc + intcpt;
1029 break;
1030 }
1031 case 1: {
1032 double phi10 = (fieldc + wiltp) / 2;
1033 wsPhoto = 1 / (1 + exp((phi10 - awc) / phi1));
1034 break;
1035 }
1036 case 2: {
1037 double slp = (1 - wiltp) / (fieldc - wiltp);
1038 double intcpt = 1 - fieldc * slp;
1039 double theta = slp * awc + intcpt;
1040 wsPhoto = (1 - exp(-2.5 * (theta - wiltp)/(1 - wiltp))) / (1 - exp(-2.5));
1041 break;
1042 }
1043 case 3:
1044 wsPhoto = 1;
1045 break;
1046 default:
1047 wsPhoto = 0;
1048 }
1049
1050 // wsPhoto can be mathematically lower than zero in some cases but
1051 // I should prevent that:
1052 if (wsPhoto <= 0) {
1053 wsPhoto = 1e-10;
1054 }
1055
1056 // Apparently wsPhoto can be greater than 1.
1057 if (wsPhoto > 1) wsPhoto = 1;
1058
1059 return wsPhoto;
1060}

1072struct ws_str watstr(double precipit, double evapo, double cws, double soildepth, double fieldc,
1073 double wiltp, double soil_saturation_capacity, double sand,
1074 double Ks, double air_entry, double b)
1075{
1076 constexpr double g = 9.8; // m / s^2 ## http://en.wikipedia.org/wiki/Standard_gravity
1077
1078 double precipM = precipit * 1e-3; /* convert precip in mm to m*/
1079 double soil_water_fraction = precipM / soildepth + cws; // m^3 m^-3
1080
1081 double runoff = 0.0;
1082 double Nleach = 0.0; /* Nleach is the NO3 leached. */
1083
1084 if (soil_water_fraction > soil_saturation_capacity) {
1085 runoff = (soil_water_fraction - soil_saturation_capacity) * soildepth; /* This is in meters */
1086 /* Here runoff is interpreted as water content exceeding saturation level */
1087 /* Need to convert to units used in the Parton et al 1988 paper. */
1088 /* The data come in mm/hr and need to be in cm/month */
1089 Nleach = runoff / (1e3 * physical_constants::molar_mass_of_water) * (0.2 + 0.7 * sand);
1090 soil_water_fraction = soil_saturation_capacity;
1091 }
1092
1093 /* The density of water is 0.9982 Mg / m^3 at 20 degrees Celsius. */
1094 /* evapo is demanded water (Mg / ha) */
1095 double npaw = soil_water_fraction - wiltp - evapo / 0.9982 / 1e4 / soildepth; // fraction of saturation.
1096
1097 /* If demand exceeds supply, the crop is getting close to wilting point and transpiration
1098 will be over estimated. In this one layer model though, the crop is practically dead. */
1099 if (npaw < 0) npaw = 0.0;
1100 double awc = npaw + wiltp;
1101
1102 /* Calculating the soil water potential based on equations from Norman and Campbell */
1103 /* tmp.psim = soTexS.air_entry * pow((awc/soTexS.fieldc*1.1),-soTexS.b); */
1104 /* New version of the soil water potential is based on
1105 * "Dynamic Simulation of Water Deficit Effects upon Maize Yield" R. F. Grant Agricultural Systems. 33(1990) 13-39. */
1106 ws_str tmp;
1107 tmp.psim = -exp(log(0.033) + (log(fieldc) - log(awc)) / (log(fieldc) - log(wiltp)) * (log(1.5) - log(0.033))) * 1e3; /* This last term converts from MPa to kPa */
1108
1109 /* This is drainage */
1110 if (awc > fieldc) {
1111 double K_psim = Ks * pow((air_entry / tmp.psim), 2 + 3 / b); /* This is hydraulic conductivity */
1112
1113 double J_w = -K_psim * (-tmp.psim / (soildepth * 0.5)) - g * K_psim; /* Campbell, pg 129 do not ignore the graviational effect. I multiply soil depth by 0.5 to calculate the average depth */
1114 double drainage = J_w * 3600 * 0.9982 * 1e-3; /* This is flow in m^3 / (m^2 * hr). */
1115 awc = awc + drainage / soildepth;
1116 }
1117
1118 tmp.awc = awc;
1119 tmp.runoff = runoff;
1120 tmp.Nleach = Nleach;
1121 return tmp;
1122}

1125 future this could be coupled with Campbell (BASIC) ideas to
1126 esitmate water potential. */
1127struct soilML_str soilML(double precipit, double transp, double *cws, double soildepth, double *depths,
1128 double soil_field_capacity, double soil_wilting_point, double soil_saturation_capacity, double soil_air_entry, double soil_saturated_conductivity,
1129 double soil_b_coefficient, double soil_sand_content, double phi1, double phi2, int wsFun,
1130 int layers, double rootDB, double LAI, double k, double AirTemp,
1131 double IRad, double winds, double RelH, int hydrDist, double rfl,
1132 double rsec, double rsdf, double soil_clod_size, double soil_reflectance, double soil_transmission,
1133 double specific_heat_of_air, double par_energy_content)
1134{
1135 constexpr double g = 9.8; /* m / s-2 ## http://en.wikipedia.org/wiki/Standard_gravity */
1136
1137 soilML_str return_value;
1138
1139 /* Crude empirical relationship between root biomass and rooting depth*/
1140 double rootDepth = fmin(rootDB * rsdf, soildepth);
1141
1142 rd_str root_distribution = rootDist(layers, rootDepth, &depths[0], rfl);
1143
1144 /* unit conversion for precip */
1145 double oldWaterIn = 0.0;
1146 double waterIn = precipit * 1e-3; /* convert precip in mm to m */
1147
1148 double drainage = 0.0;
1149 double wsPhotoCol = 0.0;
1150 double LeafWSCol = 0.0;
1151 double Sevap = 0.0;
1152 double oldEvapoTra = 0.0;
1153
1154 for (int i = layers - 1; i >= 0; --i) { /* The counter, i, decreases because I increase the water content due to precipitation in the last layer first*/
1155
1156 double layerDepth = depths[i+1] - depths[i]; /* This supports unequal depths. */
1157
1158 if (hydrDist > 0) {
1159 /* For this section see Campbell and Norman "Environmental BioPhysics" Chapter 9*/
1160 /* First compute the matric potential */
1161 double psim1 = soil_air_entry * pow((cws[i]/soil_saturation_capacity), -soil_b_coefficient); /* This is matric potential of current layer */
1162 double dPsim;
1163 if (i > 0) {
1164 double psim2 = soil_air_entry * pow((cws[i-1]/soil_saturation_capacity), -soil_b_coefficient); /* This is matric potential of next layer */
1165 dPsim = psim1 - psim2;
1166 /* The substraction is from the layer i - (i-1). If this last term is positive then it will move upwards. If it is negative it will move downwards. Presumably this term is almost always positive. */
1167 } else {
1168 dPsim = 0.0;
1169 }
1170
1171 double K_psim = soil_saturated_conductivity * pow((soil_air_entry/psim1), 2+3/soil_b_coefficient); /* This is hydraulic conductivity */
1172 double J_w = K_psim * (dPsim/layerDepth) - g * K_psim; /* Campbell, pg 129 do not ignore the graviational effect*/
1173 /* Notice that K_psim is positive because my reference system is reversed */
1174 /* This last result should be in kg/(m2 * s) */
1175 J_w *= 3600 * 0.9882 * 1e-3; /* This is flow in m3 / (m^2 * hr). */
1176
1177 if (i == (layers-1) && J_w < 0) {
1178 /* cws[i] = cws[i] + J_w / layerDepth; Although this should be done it drains the last layer too much.*/
1179 drainage += J_w;
1180 } else {
1181 cws[i] = cws[i] - J_w / layerDepth;
1182 if (i > 0) {
1183 cws[i - 1] = cws[i - 1] + J_w / layerDepth;
1184 }
1185 }
1186 }
1187
1188 if (cws[i] > soil_saturation_capacity) cws[i] = soil_saturation_capacity;
1189 if (cws[i] < soil_wilting_point) cws[i] = soil_wilting_point;
1190 // Here is a convention aw is available water in volume and awc
1191 // is available water content as a fraction of the soil section being investigated.
1192 double aw = cws[i] * layerDepth;
1193 /* Available water (for this layer) is the current water status times the layer depth */
1194
1195 if (waterIn > 0) {
1196 /* There is some rain. Need to add it.*/
1197 aw += waterIn / layers + oldWaterIn; /* They are both in meters so it works */
1198 /* Adding the same amount to water to each layer */
1199 /* In case there is overflow */
1200 double diffw = soil_field_capacity * layerDepth - aw;
1201
1202 if (diffw < 0) {
1203 /* This means that precipitation exceeded the capacity of the first layer */
1204 /* Save this amount of water for the next layer */
1205 oldWaterIn = -diffw;
1206 aw = soil_field_capacity * layerDepth;
1207 } else {
1208 oldWaterIn = 0.0;
1209 }
1210 }
1211
1212 /* Root Biomass */
1213 double rootATdepth = rootDB * root_distribution.rootDist[i];
1214 return_value.rootDist[i] = rootATdepth;
1215 /* Plant available water is only between current water status and permanent wilting point */
1216 /* Plant available water */
1217 double pawha = (aw - soil_wilting_point * layerDepth) * 1e4;
1218
1219 if (pawha < 0) pawha = 0;
1220
1221 double Ctransp = 0.0;
1222 double EvapoTra = 0.0;
1223 double Newpawha;
1224
1225 if (i == 0) {
1226 /* Only the first layer is affected by soil evaporation */
1227 double awc2 = aw / layerDepth;
1228 /* SoilEvapo function needs soil water content */
1229 Sevap = SoilEvapo(
1230 LAI, k, AirTemp, IRad, awc2, soil_field_capacity,
1231 soil_wilting_point, winds, RelH, rsec, soil_clod_size,
1232 soil_reflectance, soil_transmission, specific_heat_of_air,
1233 par_energy_content) * 3600 * 1e-3 * 10000; // Mg / ha / hr. 3600 s / hr * 1e-3 Mg / kg * 10000 m^2 / ha.
1234 /* I assume that crop transpiration is distributed simlarly to
1235 root density. In other words the crop takes up water proportionally
1236 to the amount of root in each respective layer.*/
1237 Ctransp = transp * root_distribution.rootDist[0];
1238 EvapoTra = Ctransp + Sevap;
1239 constexpr double density_of_water_at_20_celcius = 0.9982; // Mg m^-3.
1240 Newpawha = pawha - EvapoTra / density_of_water_at_20_celcius;
1241 /* The first term in the rhs pawha is the m3 of water available in this layer.
1242 EvapoTra is the Mg H2O ha-1 of transpired and evaporated water. 1/0.9882 converts from Mg to m3 */
1243 } else {
1244 Ctransp = transp * root_distribution.rootDist[i];
1245 EvapoTra = Ctransp;
1246 Newpawha = pawha - (EvapoTra + oldEvapoTra);
1247 }
1248
1249 if (Newpawha < 0) {
1250 /* If the Demand is not satisfied by this layer. This will be stored and added to subsequent layers*/
1251 oldEvapoTra = -Newpawha;
1252 aw = soil_wilting_point * layerDepth;
1253 }
1254
1255 double awc = Newpawha / 1e4 / layerDepth + soil_wilting_point;
1256
1257 /* This might look like a weird place to populate the structure, but is more convenient*/
1258 return_value.cws[i] = awc;
1259
1260 // To-do: Replace this block with a call to compute_wsPhoto.
1261 /* three different type of equations for modeling the effect of water stress on vmax and leaf area expansion.
1262 The equation for leaf area expansion is more severe than the one for vmax. */
1263 double wsPhoto = 0.0;
1264 double slp = 0.0;
1265 double intcpt = 0.0;
1266 double theta = 0.0;
1267
1268 if (wsFun == 0) { /* linear */
1269 slp = 1/(soil_field_capacity - soil_wilting_point);
1270 intcpt = 1 - soil_field_capacity * slp;
1271 wsPhoto = slp * awc + intcpt;
1272 } else if (wsFun == 1) {
1273 double phi10 = (soil_field_capacity + soil_wilting_point)/2;
1274 wsPhoto = 1/(1 + exp((phi10 - awc)/ phi1));
1275 } else if (wsFun == 2) {
1276 slp = (1 - soil_wilting_point)/(soil_field_capacity - soil_wilting_point);
1277 intcpt = 1 - soil_field_capacity * slp;
1278 theta = slp * awc + intcpt;
1279 wsPhoto = (1 - exp(-2.5 * (theta - soil_wilting_point)/(1 - soil_wilting_point))) / (1 - exp(-2.5));
1280 } else if (wsFun == 3) {
1281 wsPhoto = 1;
1282 }
1283
1284 if (wsPhoto <= 0 )
1285 wsPhoto = 1e-10; /* This can be mathematically lower than zero in some cases but I should prevent that. */
1286
1287 wsPhotoCol += wsPhoto;
1288
1289 double LeafWS = pow(awc, phi2) * 1/pow(soil_field_capacity, phi2);
1290 if (wsFun == 3) {
1291 LeafWS = 1;
1292 }
1293 LeafWSCol += LeafWS;
1294
1295 }
1296
1297 if (waterIn > 0) {
1298 drainage = waterIn;
1299 /* Need to convert to units used in the Parton et al 1988 paper. */
1300 /* The data comes in mm/hr and it needs to be in cm/month */
1301 return_value.Nleach = drainage * 0.1 * (1/24 * 30) / (1e3 * physical_constants::molar_mass_of_water * (0.2 + 0.7 * soil_sand_content));
1302 }
1303 else {
1304 return_value.Nleach = 0.0;
1305 }
1306
1307
1308 return_value.rcoefPhoto = (wsPhotoCol/layers);
1309 return_value.drainage = drainage;
1310 return_value.rcoefSpleaf = (LeafWSCol/layers);
1311 return_value.SoilEvapo = Sevap;
1312
1313 return return_value;
1314}

1433struct rd_str rootDist(int n_layers, double rootDepth, double *depths, double rfl)
1434{
1435 /*
1436 * Calculate the fraction of total root mass for each layer in `depths` assuming the mass
1437 * is follows a Poisson distribution along the depth.
1438 *
1439 * Returns a struct with one array of size `n_layers`.
1440 * Each element in the array is the fraction of total root mass in that layer.
1441 * The sum of all elements of the result equals 1.
1442 */
1443
1444 double layerDepth = 0.0;
1445 double CumLayerDepth = 0.0;
1446 int CumRootDist = 1;
1447 double rootDist[n_layers];
1448 double cumulative_a = 0.0;
1449
1450 for (int i = 0; i < n_layers; ++i) {
1451 if (i == 0) {
1452 layerDepth = depths[1];
1453 } else {
1454 layerDepth = depths[i] - depths[i-1];
1455 }
1456
1457 CumLayerDepth += layerDepth;
1458
1459 if (rootDepth > CumLayerDepth) {
1460 CumRootDist++;
1461 }
1462 }
1463
1464 for (int j = 0; j < n_layers; ++j) {
1465 if (j < CumRootDist) {
1466 double a = poisson_density(j + 1, (double)CumRootDist * rfl);
1467 rootDist[j] = a;
1468 cumulative_a += a;
1469 } else {
1470 rootDist[j] = 0;
1471 }
1472 }
1473
1474 struct rd_str result;
1475 for (int k = 0; k < n_layers; ++k) {
1476 result.rootDist[k] = rootDist[k] / cumulative_a;
1477 }
1478 return result;
1479}

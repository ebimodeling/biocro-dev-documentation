179inline double utilization_hill(double KM, double n, double concentration)
180{
181 return pow(concentration, n) / (pow(KM, n) + pow(concentration, n));
182}

184void utilization_growth_calculator::do_operation() const
185{
186 // Collect inputs and make calculations
187
188 //double dawn_phase = *dawn_phase_ip;
189 //if(dawn_phase > 3.14159265358979323846) dawn_phase -= 2*3.14159265358979323846;
190 //double rate_factor = exp(-(dawn_phase + 1) * (dawn_phase + 1));
191
192 double grain_TTc = *grain_TTc_ip;
193 double TTc = *TTc_ip;
194
195 double kLeaf = *rate_constant_leaf_ip;
196 double kStem = *rate_constant_stem_ip;
197 double kRoot = *rate_constant_root_ip;
198 double kRhizome = *rate_constant_rhizome_ip;
199 double kGrain = *rate_constant_grain_ip;
200
201 /*
202 double kLeaf = *rate_constant_leaf_ip * rate_factor;
203 double kStem = *rate_constant_stem_ip * rate_factor;
204 double kRoot = *rate_constant_root_ip * rate_factor;
205 double kRhizome = *rate_constant_rhizome_ip * rate_factor;
206 double kGrain = *rate_constant_grain_ip * rate_factor;
207 */
208
209 double KmLeaf = *KmLeaf_ip;
210 double KmStem = *KmStem_ip;
211 double KmRoot = *KmRoot_ip;
212 double KmRhizome = *KmRhizome_ip;
213 double KmGrain = *KmGrain_ip;
214
215 double resistance_leaf_to_stem = *resistance_leaf_to_stem_ip;
216 double resistance_stem_to_grain = *resistance_stem_to_grain_ip;
217 double resistance_stem_to_root = *resistance_stem_to_root_ip;
218 double resistance_stem_to_rhizome = *resistance_stem_to_rhizome_ip;
219
220 double Leaf = *Leaf_ip;
221 double Stem = *Stem_ip;
222 double Root = *Root_ip;
223 double Grain = *Grain_ip;
224 double Rhizome = *Rhizome_ip;
225
226 double substrate_pool_leaf = *substrate_pool_leaf_ip;
227 double substrate_pool_stem = *substrate_pool_stem_ip;
228 double substrate_pool_root = *substrate_pool_root_ip;
229 double substrate_pool_rhizome = *substrate_pool_rhizome_ip;
230 double substrate_pool_grain = *substrate_pool_grain_ip;
231
232 // Create a logistic scaling coefficient to smoothly ramp up the grain growth rate
233 // The midpoint is a TTc value defined in the input quantities
234 // The maximum value is 1
235 // The steepness is chosen so the coefficient reaches 95% of its maximum
236 // value when TTc passes the midpoint by 8 thermal days:
237 // k = ln(0.95/(1-0.95))/8 = 0.368
238 // This roughly corresponds to grain growth "turning on" over the course of
239 // 2*8 = 16 thermal days
240
241 double kGrain_scale = 1 / (1 + exp(-0.368 * (TTc - grain_TTc)));
242
243 // Create an exponential scaling coefficient to smoothly ramp down the rhizome
244 // "retranslocation" rate
245
246 //double kRhizome_retrans = 0.01 * kRhizome * exp(-TTc/300.0);
247 double kRhizome_retrans = 0.0;
248
249 // Calculate the substrate utilization rates
250 // mass_fraction_leaf = substrate_pool_leaf / Leaf is a measure of the substrate concentration in the leaf,
251 // made under the assumption that leaf volume is proportional to its structural carbon component, i.e., Leaf
252 // We assume there is a catalyzed reaction transforming the substrate carbon into structural carbon.
253 // Using Michaelis-Menton kinetics, the rate of change of the structural carbon concentration is proportional
254 // to [substrate]/(Km + [substrate]), i.e., mass_fraction_leaf / (KmLeaf + mass_fraction_leaf)
255 // The total substrate utilization is this rate multiplied by the leaf volume, i.e.
256 // kLeaf * Leaf * mass_fraction_leaf / (KmLeaf + mass_fraction_leaf) = kLeaf * substrate_pool_leaf / (KmLeaf + mass_fraction_leaf)
257 // where kLeaf includes the various proportionality constants mentioned previously
258 // See Thornley, J. H. M. A Model to Describe the Partitioning of Photosynthate during Vegetative Plant Growth. Ann Bot 36, 419–430 (1972)
259 // Alternatively, we could use a Hill equation to model utilization. This may be especially helpful for the leaf, since utilization
260 // can be made to fall off faster at small substrate concentrations. Note that the M-M equation is a special case of the Hill
261 // equation when n = 1.
262
263 double n = 2.0; // Set to 1.0 for Michaelis-Morley kinetics. Use larger values to get a steeper utilization cutoff at small substrate concentrations.
264
265 double mass_fraction_leaf = substrate_pool_leaf / Leaf;
266 double utilization_leaf = kLeaf * Leaf * utilization_hill(KmLeaf, n, mass_fraction_leaf);
267
268 double mass_fraction_stem = substrate_pool_stem / Stem;
269 double utilization_stem = kStem * Stem * utilization_hill(KmStem, n, mass_fraction_stem);
270
271 double mass_fraction_root = substrate_pool_root / Root;
272 double utilization_root = kRoot * Root * utilization_hill(KmRoot, n, mass_fraction_root);
273
274 double mass_fraction_rhizome = substrate_pool_rhizome / Rhizome;
275 double utilization_rhizome = kRhizome * Rhizome * utilization_hill(KmRhizome, n, mass_fraction_rhizome);
276 utilization_rhizome -= kRhizome_retrans * Rhizome; // Add an additional term that allows the rhizome to redistribute some of its mass in the early stages of growth, i.e., when TTc is small
277
278 double mass_fraction_grain = substrate_pool_grain / Grain;
279 double utilization_grain = kGrain * kGrain_scale * Grain * utilization_hill(KmGrain, n, mass_fraction_grain);
280
281 // Calculate the transport rates
282 // We make the simple assumption that substrate transport is proportional to a concentration gradient between tissues
283 // and inversely proportional to a resistance, analogous to current in an electrical circuit
284 // It is important to scale the transport as the plant grows. For example, if the plant uniformly doubles in volume,
285 // the roots now require twice as much carbon substrate to continue growth at the same rate, so the resistance between
286 // the stem and root should be halved (or transport should double). We accomplish this scaling with the beta factor.
287 // See Thornley, J. H. M. A Model to Describe the Partitioning of Photosynthate during Vegetative Plant Growth. Ann Bot 36, 419–430 (1972)
288
289 double beta = Leaf + Grain + Stem + Root + Rhizome;
290
291 double transport_leaf_to_stem = beta * (mass_fraction_leaf - mass_fraction_stem) / resistance_leaf_to_stem;
292 double transport_stem_to_grain = beta * (mass_fraction_stem - mass_fraction_grain) / resistance_stem_to_grain;
293 double transport_stem_to_root = beta * (mass_fraction_stem - mass_fraction_root) / resistance_stem_to_root;
294 double transport_stem_to_rhizome = beta * (mass_fraction_stem - mass_fraction_rhizome) / resistance_stem_to_rhizome;
295
296 // Update the output quantity list
297 update(utilization_leaf_op, utilization_leaf);
298 update(utilization_stem_op, utilization_stem);
299 update(utilization_root_op, utilization_root);
300 update(utilization_rhizome_op, utilization_rhizome);
301 update(utilization_grain_op, utilization_grain);
302
303 update(transport_leaf_to_stem_op, transport_leaf_to_stem);
304 update(transport_stem_to_grain_op, transport_stem_to_grain);
305 update(transport_stem_to_root_op, transport_stem_to_root);
306 update(transport_stem_to_rhizome_op, transport_stem_to_rhizome);
307
308 update(mass_fraction_leaf_op, mass_fraction_leaf);
309 update(mass_fraction_stem_op, mass_fraction_stem);
310 update(mass_fraction_root_op, mass_fraction_root);
311 update(mass_fraction_rhizome_op, mass_fraction_rhizome);
312 update(mass_fraction_grain_op, mass_fraction_grain);
313
314 update(kGrain_scale_op, kGrain_scale);
315}

390 // solar reaches ~150, which is well below the max on a typical day
391 double L = 1.0 / (1.0 + exp(-0.058 * (solar - 100.0)));

393 // Calculate some parameters defined in the Pokhilko model
394 double Gn = p28 * GI_prot_cytoplasm / (p29 + m19 + p17 * ELF3_nuclear);
395 double EGn = (p18 * GI_ELF3_cytoplasm + p17 * ELF3_nuclear * Gn) / (m9 * COP1_nuclear_night + m10 * COP1_nuclear_day + p31);

94void FvCB::do_operation() const
95{
96 // Collect input quantities and make calculations
97
98 double Ci = *Ci_ip;
99 double Gstar = *Gstar_ip;
100 double Vcmax = *Vcmax_ip;
101 double Kc = *Kc_ip;
102 double O2 = *O2_ip;
103 double Ko = *Ko_ip;
104 double J = *J_ip;
105 double electrons_per_carboxylation = *electrons_per_carboxylation_ip;
106 double electrons_per_oxygenation = *electrons_per_oxygenation_ip;
107 double maximum_tpu_rate = *maximum_tpu_rate_ip;
108 double Rd = *Rd_ip;
109 double Ca = *Ca_ip;
110 double atmospheric_pressure = *atmospheric_pressure_ip;
111 double leaf_stomatal_conductance = *leaf_stomatal_conductance_ip;
112
113 double rubisco_limited = Vcmax * (Ci - Gstar) / (Ci + Kc * (1.0 + O2 / Ko));
114
115 double rubp_limited = J * (Ci - Gstar) / (electrons_per_carboxylation * Ci + 2.0 * electrons_per_oxygenation * Gstar);
116 rubp_limited = std::max(rubp_limited, 0.0); // This rate can't be negative.
117
118 double tpu_limited = 3.0 * maximum_tpu_rate / (1.0 - Gstar / Ci);
119
120 double carboxylation_rate = std::min(rubisco_limited, std::min(rubp_limited, tpu_limited)); // The overall carboxylation rate is the rate of the slowest process.
121 double net_assimilation_rate = carboxylation_rate - Rd;
122
123 Ci = Ca - net_assimilation_rate * 1.6 * atmospheric_pressure / leaf_stomatal_conductance;
124
125 // Update the output quantity list
126 update(carboxylation_rate_op, carboxylation_rate);
127 update(net_assimilation_rate_op, net_assimilation_rate);
128 update(Ci_op, Ci);
129}

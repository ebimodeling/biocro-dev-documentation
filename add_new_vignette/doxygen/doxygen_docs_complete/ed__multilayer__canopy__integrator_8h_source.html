139void ed_multilayer_canopy_integrator::run() const
140{
141 const double LAIc = lai / nlayers;
142 double canopy_assimilation_rate = 0;
143 double canopy_transpiration_rate = 0;
144 double canopy_conductance = 0;
145 double GrossAssim = 0;
146
147 // Integrate assimilation, transpiration, and conductance throughout the
148 // canopy
149 for (int i = 0; i < nlayers; ++i) {
150 const double sunlit_lai = *sunlit_fraction_ips[i] * LAIc;
151 const double shaded_lai = *shaded_fraction_ips[i] * LAIc;
152
153 canopy_assimilation_rate +=
154 *sunlit_assimilation_net_ips[i] * sunlit_lai +
155 *shaded_assimilation_net_ips[i] * shaded_lai;
156
157 canopy_transpiration_rate +=
158 *sunlit_transpiration_rate_ips[i] * sunlit_lai +
159 *shaded_transpiration_rate_ips[i] * shaded_lai;
160
161 canopy_conductance +=
162 *sunlit_conductance_stomatal_h2o_ips[i] * sunlit_lai +
163 *shaded_conductance_stomatal_h2o_ips[i] * shaded_lai;
164
165 GrossAssim +=
166 *sunlit_assimilation_gross_ips[i] * sunlit_lai +
167 *shaded_assimilation_gross_ips[i] * shaded_lai;
168 }
169
170 // Modify net assimilation to account for respiration
171 // Note: this was originally only done for the C3 canopy
172 // Note: it seems like this should not be necessary since the assimilation
173 // model includes respiration
174 canopy_assimilation_rate *= (1.0 - growth_respiration_fraction);
175
176 // For assimilation, we need to convert mol / m^2 / s into Mg / ha / hr,
177 // assuming that all carbon is converted into biomass in the form of glucose
178 // (C6H12O6), i.e., six assimilated CO2 molecules contribute one glucose
179 // molecule. Using the molar mass of glucose in kg / mol, the conversion can
180 // be accomplished with the following factor:
181 // (1 glucose / 6 CO2) * (3600 s / hr) * (1e-3 Mg / kg) * (1e4 m^2 / ha)
182 // = 6e3 s * Mg * m^2 / (hr * kg * ha)
183 const double cf = physical_constants::molar_mass_of_glucose * 6e3; // (Mg / ha / hr) / (mol / m^2 / s)
184
185 // For transpiration, we need to convert mol / m^2 / s into Mg / ha / hr
186 // using the molar mass of water in kg / mol, which can be accomplished by
187 // the following conversion factor:
188 // (3600 s / hr) * (1e-3 Mg / kg) * (1e4 m^2 / ha)
189 // = 36e3 s * Mg * m^2 / (hr * kg * ha)
190 const double cf2 = physical_constants::molar_mass_of_water * 36e3; // (Mg / ha / hr) / (mol / m^2 / s)
191
192 // Convert conductance values from mol / m^2 / s to mmol / m^2 / s using 1e3 mmol / mol
193
194 update(canopy_assimilation_rate_op, canopy_assimilation_rate * cf);
195 update(GrossAssim_op, GrossAssim * cf);
196 update(canopy_transpiration_rate_op, canopy_transpiration_rate * cf2);
197 update(canopy_conductance_op, canopy_conductance * 1e3);
198}

47std::vector<double> random_vector_on_n_dimensional_sphere(size_t n)
48{
49 // Create a vector and populate it with normally distributed values
50 std::vector<double> vec(n);
51 for (size_t i = 0; i < n; ++i) {
52 vec[i] = normal_rand();
53 }
54
55 // Find its norm
56 double vec_norm = sqrt(std::inner_product(vec.begin(), vec.end(), vec.begin(), 0.0));
57
58 // Check for the rare case that we have somehow chosen the zero vector
59 // In this case, just set all elements to 1 and recalculate the norm
60 if (vec_norm < calculation_constants::eps_zero) {
61 for (size_t i = 0; i < n; ++i) {
62 vec[i] = 1.0;
63 }
64 vec_norm = sqrt(std::inner_product(vec.begin(), vec.end(), vec.begin(), 0.0));
65 }

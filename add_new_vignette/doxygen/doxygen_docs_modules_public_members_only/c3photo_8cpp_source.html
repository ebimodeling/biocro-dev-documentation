57 const quantity<flux> maximum_tpu_rate = _TPU_rate_max * 1e-6 * mole / square_meter / second;
58
59 /* Temperature corrections are from the following sources:
60 Bernacchi et al. (2003) Plant, Cell and Environment, 26(9), 1419-1430.
61 https://doi.org/10.1046/j.0016-8025.2003.01050.x
62 Bernacchi et al. (2001) Plant, Cell and Environment, 24(2), 253-259.
63 https://doi.org/10.1111/j.1365-3040.2001.00668.x */
64 /* Note: Values in Dubois and Bernacchi are incorrect. */
65 const quantity<mole_fraction> Kc = 1e-6 * arrhenius_exponent(38.05, 79.43e3 * joule / mole, leaf_temperature);
66 const quantity<mole_fraction> Ko = 1e-3 * arrhenius_exponent(20.30, 36.38e3 * joule / mole, leaf_temperature);
67 const quantity<mole_fraction> Gstar = 1e-6 * arrhenius_exponent(19.02, 37.83e3 * joule / mole, leaf_temperature);
68 const quantity<flux> Vcmax = Vcmax0 * arrhenius_exponent(26.35, 65.33e3 * joule / mole, leaf_temperature);
69 const quantity<flux> Jmax = Jmax0 * arrhenius_exponent(17.57, 43.54e3 * joule / mole, leaf_temperature);
70 const quantity<flux> Rd = Rd0 * arrhenius_exponent(18.72, 46.39e3 * joule / mole, leaf_temperature);
71
72 const double leaf_temperature_celsius = leaf_temperature.value() - conversion_constants::celsius_to_kelvin;
73 const quantity<dimensionless> theta =
74 thet + 0.018 * leaf_temperature_celsius - 3.7e-4 * pow(leaf_temperature_celsius, 2);
75
76 /* Light limited */
77 const quantity<dimensionless> dark_adapted_phi_PSII =
78 0.352 + 0.022 * leaf_temperature_celsius -
79 3.4 * pow(leaf_temperature_celsius, 2) / 10000; // Bernacchi et al. (2003). See reference above.
80 const quantity<flux> I2 = Qp * dark_adapted_phi_PSII * (1.0 - leaf_reflectance) / 2.0;
81
82 const quantity<flux> J = (Jmax + I2 - root<2>(pow<2>(Jmax + I2) - 4.0 * theta * I2 * Jmax)) / (2.0 * theta);

101 const double TPU_c = 25.5; // dimensionless. fitted constant
102 const quantity<energy_over_amount> Ha = 62.99e3 * joule / mole; // enthalpy of activation
103 const quantity<energy_over_temperature_amount> S = 0.588e3 * joule / kelvin / mole; // entropy
104 const quantity<energy_over_amount> Hd = 182.14e3 * joule / mole; // enthalpy of deactivation
105 const quantity<energy_over_temperature_amount> R = physical_constants::ideal_gas_constant * joule / kelvin / mole; // gas constant
106 const quantity<dimensionless> top = leaf_temperature.value() * arrhenius_exponent(TPU_c, Ha, leaf_temperature);
107 const quantity<dimensionless> bot = 1.0 + arrhenius_exponent(S / R, Hd, leaf_temperature);
108 double TPU_rate_scaler = top.value() / bot.value(); // dimensionless
109
110 /* In Yang et al. the equation for `TPU_rate_scaler` is described as
111 producing the TPU rate relative to 25 degrees C. However, it does not do
112 that. Here we calculate the value of that equation at 25 degrees C, then
113 divide `TPU_rate_scaler` by that value to get a rate relative to 25
114 degrees C. */
115 const double TPU_rate_scaler25 = 306.742; // dimensionless. This is `top / bottom` at 25 degrees C.
116 TPU_rate_scaler /= TPU_rate_scaler25; // dimensionless. Normalize to 25 degrees C.
117
118 /* The alpha constant for calculating Ap is from
119 Eq. 2.26, von Caemmerer, S. Biochemical models of leaf photosynthesis.*/
120 double alpha_TPU = 0.0; // dimensionless. Without more information, alpha=0 is often assumed.
121
122 int iterCounter = 0;
123 int max_iter = 1000;
124 while (iterCounter < max_iter) {
125 quantity<flux> OldAssim = co2_assimilation_rate;
126
127 /* Rubisco limited carboxylation */
128 Ci = Ci_pa / atmospheric_pressure;
129 quantity<flux> Ac1 = Vcmax * (Ci - Gstar);
130 double Ac2 = Ci + Kc * (1 + Oi / Ko);
131 quantity<flux> Ac = Ac1 / Ac2;
132
133 /* Light limited portion */
134 quantity<flux> Aj1 = J * (Ci - Gstar);
135 double Aj2 = electrons_per_carboxylation * Ci + 2.0 * electrons_per_oxygenation * Gstar;
136 quantity<flux> Aj = Aj1 / Aj2;
137 if (Aj < 0.0 * mole / square_meter / second) {
138 Aj = 0.0 * mole / square_meter / second;
139 }
140
141 /* Triose phosphate utilization limited */
142 quantity<flux> Ap = 3.0 * maximum_tpu_rate * (Ci - Gstar) / (Ci - (1.0 + 1.5 * alpha_TPU) * Gstar);
143 Ap = Ap * TPU_rate_scaler;
144
145 if (Ac < Aj && Ac < Ap) {
146 Vc = Ac;
147 } else if (Aj < Ac && Aj < Ap) {
148 Vc = Aj;
149 } else if (Ap < Ac && Ap < Aj) {
150 if (Ap < 0 * mole / square_meter / second) {
151 Ap = 0 * mole / square_meter / second;
152 }
153 Vc = Ap;
154 }
155
156 co2_assimilation_rate = Vc - Rd;
157
158 if (water_stress_approach == 0) {
159 co2_assimilation_rate *= quantity<dimensionless>(StomWS);
160 }
161
162 Gs = ball_berry(co2_assimilation_rate.value(), Ca * 1e-6, RH, bb0, bb1) * 1e-3 * mole / square_meter / second;
163
164 if (water_stress_approach == 1) {
165 Gs = Gsw_min + StomWS * (Gs - Gsw_min);
166 }
167
168 if (Gs <= 0 * mole / square_meter / second) {
169 Gs = 1e-5 * 1e-3 * mole / square_meter / second;
170 }
171
172 Ci_pa = Ca_pa - co2_assimilation_rate * 1.6 * atmospheric_pressure / Gs;
173
174 if (Ci_pa < 0 * pascal) {
175 Ci_pa = 1e-5 * pascal;
176 }
177
178 if (abs(OldAssim - co2_assimilation_rate) < Tol) {
179 break;
180 }
181
182 ++iterCounter;
183 }
184
185 struct c3_str result;
186 result.Assim = co2_assimilation_rate.value() * 1e6; // micromole / m^2 / s.
187 result.Gs = Gs.value() * 1e3; // mmol / m^2 / s.
188 result.Ci = Ci.value() * 1e6; // micromole / mol.
189 result.GrossAssim = (co2_assimilation_rate.value() + Rd.value()) * 1e6; // micromole / m^2 / s.
190 return result;
191}

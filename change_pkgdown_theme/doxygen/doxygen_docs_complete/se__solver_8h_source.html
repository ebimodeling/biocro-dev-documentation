129template <typename observer_type>
130bool se_solver::solve(
131 std::unique_ptr<simultaneous_equations> const& se,
132 std::vector<double> const& initial_guess_for_root,
133 std::vector<double> const& lower_bounds,
134 std::vector<double> const& upper_bounds,
135 std::vector<double> const& absolute_error_tolerances,
136 std::vector<double> const& relative_error_tolerances,
137 std::vector<double>& final_value_for_root,
138 observer_type observer)
139{
140 // Reset all the status indicators
141 solve_method_has_been_called = true;
142 converged_abs = false;
143 converged_rel = false;
144 num_iterations = 0;
145
146 // Define a lambda to simplify error checking. If all entries in `convergence_error_vector`
147 // are `false`, `convergence_achieved` is set to true.
148 auto no_errors_occurred = [](std::vector<bool> convergence_error_vector) -> bool {
149 int num_errors = std::accumulate(convergence_error_vector.begin(), convergence_error_vector.end(), 0);
150 if (num_errors < 1) {
151 return true;
152 } else {
153 return false;
154 }
155 };
156
157 // Initialize local variables for the loop
158 std::vector<double> prev_guess = initial_guess_for_root;
159 std::vector<double> next_guess = prev_guess;
160 std::vector<double> prev_difference_vector(initial_guess_for_root.size());
161 std::vector<double> next_difference_vector(initial_guess_for_root.size());
162 bool problem_occurred_while_getting_guess = false;
163 bool zero_found = false;
164
165 // Evaluate the simultaneous equations at the first guess to get the
166 // corresponding difference vector
167 (*se)(prev_guess, prev_difference_vector); // modifies prev_difference_vector
168
169 // Run the loop until convergence is achieved or the max number of iterations is exceeded
170 do {
171 // Pass the previous guess to the observer
172 observer(prev_guess);
173
174 // Check to see if the previous guess is a zero
175 converged_abs = no_errors_occurred(has_not_converged_abs(prev_difference_vector, absolute_error_tolerances));
176 converged_rel = no_errors_occurred(has_not_converged_rel(prev_difference_vector, prev_guess, relative_error_tolerances));
177 zero_found = converged_abs && converged_rel;
178
179 // If a zero was found or a problem occurred while getting the guess, break out of the loop
180 // Otherwise, get a new guess and try again.
181 if (zero_found || problem_occurred_while_getting_guess) {
182 break;
183 } else {
184 // Get the next guess
185 problem_occurred_while_getting_guess = get_next_guess(
186 se, lower_bounds, upper_bounds,
187 prev_guess, prev_difference_vector,
188 next_guess, next_difference_vector); // modifies next_guess and next_difference_vector
189
190 // Increment the iteration counter
191 ++num_iterations;
192
193 // Update the guess and difference vector
194 prev_guess = next_guess;
195 prev_difference_vector = next_difference_vector;
196 }
197
198 } while (num_iterations < max_iterations);
199
200 final_value_for_root = prev_guess;
201
202 return zero_found;
203}

51template <typename equation_ptr_type, typename x_vector_type, typename time_type, typename f_vector_type, typename matrix_type>
52void calculate_jacobian(
53 equation_ptr_type const& equation_ptr,
54 x_vector_type const& x,
55 time_type t,
56 f_vector_type const& f_current,
57 matrix_type& jacobi)
58{
59 size_t n = x.size();
60
61 // Make a vector to store the perturbed f(x,t)
62 f_vector_type f_perturbed(n);
63
64 // Perturb each element x_i of the input vector to find df_j(x,t)/dx_i,
65 // which is stored at jacobi(j,i)
66 double h;
67 x_vector_type x_perturbed = x;
68
69 for (size_t i = 0; i < n; i++) {
70 // Detemine the step size h by taking a fraction of x[i]: h = x[i] * eps_deriv.
71 // Ensure that the step size h is close to this value but is exactly representable
72 // (see Numerical Recipes in C, 2nd ed., Section 5.7)
73 h = x[i] * calculation_constants::eps_deriv;
74 if (h == 0.0) {
75 h = calculation_constants::eps_deriv * calculation_constants::eps_deriv;
76 }
77 double temp = x[i] + h;
78 h = temp - x[i];
79
80 // Calculate the new function value
81 x_perturbed[i] = x[i] + h; // Add h to the ith differential quantity
82 evaluate_equations(equation_ptr, x_perturbed, t, f_perturbed); // Calculate f_perturbed
83
84 // Store the results in the Jacobian matrix
85 for (size_t j = 0; j < n; j++) {
86 jacobi(j, i) = (f_perturbed[j] - f_current[j]) / h;
87 }
88
89 // Reset the ith differential quantity
90 x_perturbed[i] = x[i];
91 }
92}

158template <typename equation_ptr_type, typename time_type, typename x_vector_type, typename f_vector_type>
159void calculate_time_derivative(
160 equation_ptr_type const& equation_ptr,
161 time_type max_time,
162 x_vector_type const& x,
163 time_type t,
164 f_vector_type const& f_current,
165 f_vector_type& dfdt)
166{
167 size_t n = x.size();
168
169 // Make a vector to store the perturbed f(x,t)
170 f_vector_type f_perturbed(n);
171
172 // Perturb the time to find df(x,t)/dt
173 // Use a forward step whenever possible
174 // Detemine the step size h by taking a fraction of t: h = t * eps_deriv.
175 // Ensure that the step size h is close to this value but is exactly representable
176 // (see Numerical Recipes in C, 2nd ed., Section 5.7)
177 double h = t * calculation_constants::eps_deriv;
178 if (h == 0.0) {
179 h = calculation_constants::eps_deriv * calculation_constants::eps_deriv;
180 }
181 if (t + h <= max_time) {
182 double temp = t + h;
183 h = temp - t;
184 evaluate_equations(equation_ptr, x, t + h, f_perturbed);
185 for (size_t j = 0; j < n; j++) {
186 dfdt[j] = (f_perturbed[j] - f_current[j]) / h;
187 }
188 } else {
189 double temp = t - h;
190 h = temp + t;
191 evaluate_equations(equation_ptr, x, t - h, f_perturbed);
192 for (size_t j = 0; j < n; j++) {
193 dfdt[j] = (f_current[j] - f_perturbed[j]) / h;
194 }
195 }
196}

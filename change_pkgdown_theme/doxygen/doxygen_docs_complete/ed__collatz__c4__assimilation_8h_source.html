132void ed_collatz_c4_assimilation::do_operation() const
133{
134 // Calculate the temperature factor
135 const double temperature_factor = pow(*collatz_q10_ip, (*temperature_leaf_ip - 25.0) / 10.0);
136
137 // Calculate the carbon-limited assimilation rate
138 const double co2_rate_constant = *collatz_k_ip * temperature_factor;
139 const double assimilation_carbon_limited = co2_rate_constant * *mole_fraction_co2_intercellular_ip;
140
141 // Calculate the rubisco-limited assimilation rate
142 const double rubisco_capacity_base = *collatz_vmax_ip * temperature_factor;
143 const double rubisco_low_temp_inhibition = 1 + exp(0.3 * (*collatz_rubisco_temperature_lower_ip - *temperature_leaf_ip));
144 const double rubisco_high_temp_inhibition = 1 + exp(0.3 * (*temperature_leaf_ip - *collatz_rubisco_temperature_upper_ip));
145 const double assimilation_rubisco_limited = rubisco_capacity_base / (rubisco_low_temp_inhibition * rubisco_high_temp_inhibition);
146
147 // Calculate the light-limited assimilation rate
148 const double assimilation_light_limited = *collatz_alpha_ip * *collatz_PAR_flux_ip;
149
150 // Use the mixing parameter theta to determine the gross assimilation rate limited by rubisco and light
151 const double M_a = *collatz_theta_ip;
152 const double M_b = -1.0 * (assimilation_rubisco_limited + assimilation_light_limited);
153 const double M_c = assimilation_rubisco_limited * assimilation_light_limited;
154 const double M_root_term = M_b * M_b - 4 * M_a * M_c;
155
156 double M;
157 if (M_a == 0) {
158 M = -M_c / M_b;
159 } else {
160 const double M_plus = (-M_b + sqrt(M_root_term)) / (2 * M_a);
161 const double M_minus = (-M_b - sqrt(M_root_term)) / (2 * M_a);
162 M = std::min(M_plus, M_minus);
163 }
164
165 // Use the mixing parameter beta to determine the gross assimilation rate limited by carbon, rubisco, and light
166 const double A_a = *collatz_beta_ip;
167 const double A_b = -1.0 * (M + assimilation_carbon_limited);
168 const double A_c = M * assimilation_carbon_limited;
169 const double A_root_term = A_b * A_b - 4 * A_a * A_c;
170
171 double A;
172 if (A_a == 0) {
173 A = -A_c / A_b;
174 } else {
175 const double A_plus = (-A_b + sqrt(A_root_term)) / (2 * A_a);
176 const double A_minus = (-A_b - sqrt(A_root_term)) / (2 * A_a);
177 A = std::min(A_plus, A_minus);
178 }
179
180 // Calculate the temperature-dependent leaf respiration
181 const double respiration_base = *collatz_rd_ip * temperature_factor;
182 const double respiration_high_temp_inhibition = 1 + exp(0.3 * (*temperature_leaf_ip - 55.0));
183 const double respiration = respiration_base / respiration_high_temp_inhibition;
184
185 // Calculate net assimilation rate
186 const double assimilation_net = A - respiration;
187
188 // Apply a possible reduction accounting for the effect of water stress on the stomata
189 const double adjusted_assimilation_net = assimilation_net * *assimilation_adjustment_factor_WS_ip;
190
191 std::string message = "collatz parameters:\n";
192 char buff[128];
193
194 sprintf(buff, " temperature_factor = %e\n", temperature_factor);
195 message += std::string(buff);
196
197 sprintf(buff, " co2_rate_constant = %e\n", co2_rate_constant);
198 message += std::string(buff);
199
200 sprintf(buff, " assimilation_carbon_limited = %e\n", assimilation_carbon_limited);
201 message += std::string(buff);
202
203 sprintf(buff, " rubisco_low_temp_inhibition = %e\n", rubisco_low_temp_inhibition);
204 message += std::string(buff);
205
206 sprintf(buff, " rubisco_high_temp_inhibition = %e\n", rubisco_high_temp_inhibition);
207 message += std::string(buff);
208
209 sprintf(buff, " assimilation_rubisco_limited = %e\n", assimilation_rubisco_limited);
210 message += std::string(buff);
211
212 sprintf(buff, " respiration_base = %e\n", respiration_base);
213 message += std::string(buff);
214
215 sprintf(buff, " respiration_high_temp_inhibition = %e\n", respiration_high_temp_inhibition);
216 message += std::string(buff);
217
218 sprintf(buff, " assimilation_net = %e\n", assimilation_net);
219 message += std::string(buff);
220
221 sprintf(buff, " adjusted_assimilation_net = %e\n", adjusted_assimilation_net);
222 message += std::string(buff);
223
224 message += std::string("\n");
225 if (collatz_print) {
226 Rprintf(message.c_str());
227 }
228
229 // Check for error conditions
230 std::map<std::string, bool> errors_to_check = {
231 {"the quadratic M_a and M_b terms cannot both be zero", fabs(M_a) < calculation_constants::eps_zero && fabs(M_b) < calculation_constants::eps_zero}, // divide by zero
232 {"the quadratic M_root_term cannot be negative for M_a != 0", fabs(M_a) > calculation_constants::eps_zero && M_root_term < 0}, // imaginary sqrt
233 {"the quadratic A_a and A_b terms cannot both be zero", fabs(A_a) < calculation_constants::eps_zero && fabs(A_b) < calculation_constants::eps_zero}, // divide by zero
234 {"the quadratic A_root_term cannot be negative for A_a != 0", fabs(A_a) > calculation_constants::eps_zero && A_root_term < 0} // imaginary sqrt
235 };
236
237 check_error_conditions(errors_to_check, get_name());
238
239 // Update outputs
240 update(collatz_KT_op, co2_rate_constant);
241 update(collatz_RT_op, respiration);
242 update(collatz_VT_op, assimilation_rubisco_limited);
243 update(assimilation_carbon_limited_op, assimilation_carbon_limited);
244 update(assimilation_rubisco_limited_op, assimilation_rubisco_limited);
245 update(assimilation_light_limited_op, assimilation_light_limited);
246 update(assimilation_gross_op, A);
247 update(respiration_op, respiration);
248 update(assimilation_net_op, adjusted_assimilation_net);
249}

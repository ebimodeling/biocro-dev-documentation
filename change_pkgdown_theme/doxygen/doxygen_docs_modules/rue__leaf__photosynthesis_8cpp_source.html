18// Determine assimilation, stomatal conductance, and Ci following `c3photoC()`.
19// Here, rather than using the FvCB model for C3 photosynthesis, gross
20// assimilation is determined using a simple RUE model. Water stress is ignored.
21// Respiration and stomatal conductance are otherwise calculated using the same
22// methods as `c3photoC()`.
23struct rue_str rue_photo(
24 double Qp, // mol / m^2 / s
25 double alpha_rue, // dimensionless
26 double Tleaf, // degrees C
27 double RH, // dimensionless from Pa / Pa
28 double Rd0, // mol / m^2 / s
29 double bb0, // mol / m^2 / s
30 double bb1, // dimensionless
31 double Ca // dimensionless from mol / mol
32)
33{
34 using conversion_constants::celsius_to_kelvin; // deg. C or K
35
36 // Convert leaf temperature to Kelvin
37 double const tlk = Tleaf + celsius_to_kelvin; // K
38
39 // Calculate gross assimilation based on incident light and radiation use
40 // efficiency
41 double const ag = Qp * alpha_rue; // mol / m^2 / s
42
43 // Determine respiration from the leaf temperature using Arrhenius exponents
44 // from Bernacchi et al. (2001) Plant, Cell and Environment, 24(2), 253-259.
45 // https://doi.org/10.1111/j.1365-3040.2001.00668.x
46 double const rd = Rd0 *
47 arrhenius_exponent(
48 18.72,
49 46.39e3,
50 tlk); // mol / m^2 / s
51
52 // Determine net assimilation
53 double const an = ag - rd; // mol / m^2 / s
54
55 // Determine stomatal conductance
56 double const gs = ball_berry(an, Ca, RH, bb0, bb1) * 1e-3; // mol / m^2 / s
57
58 // Determine intercellular CO2 concentration
59 double const ci = Ca - an * 1.6 / gs; // dimensionless
60
61 // Return the results
62 struct rue_str result;
63 result.Assim = an * 1e6; // micromol / m^2 / s
64 result.Gs = gs * 1e3; // mmol / m^2 / s
65 result.Ci = ci * 1e6; // micromole / mol
66 result.GrossAssim = ag * 1e6; // micromole / m^2 / s
67 return result;
68}

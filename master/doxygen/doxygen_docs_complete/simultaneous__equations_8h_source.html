121template <typename vector_type>
122void simultaneous_equations::run_modules(vector_type const& unknown_quantity_vector)
123{
124 // Increment the counter
125 ++ncalls;
126
127 // Clear all the outputs of direct modules in the main quantity map
128 for (auto const& x : direct_ptr_pairs) {
129 *x.first = 0.0;
130 }
131
132 // Set the input values of the unknown quantities in the main quantity map
133 for (size_t i = 0; i < unknown_ptrs.size(); i++) {
134 *(unknown_ptrs[i]) = unknown_quantity_vector[i];
135 }
136
137 // Initialize the module output map so it has the same values as the main map
138 module_output_map = quantities;
139
140 // Run each module and store its output in the main quantity map
141 for (std::unique_ptr<module> const& m : direct_modules) {
142 m->run();
143 for (auto const& x : direct_ptr_pairs) {
144 *x.first = *x.second;
145 }
146 }
147}

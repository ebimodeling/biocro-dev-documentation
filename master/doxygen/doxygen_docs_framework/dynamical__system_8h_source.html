322template <class vector_type>
323void dynamical_system::run_differential_modules(vector_type& dxdt)
324{
325 // Reset the derivatives of the differential quantities
326 for (auto& x : differential_quantity_derivatives) {
327 x.second = 0.0;
328 }
329
330 // Run the modules
331 run_module_list(differential_modules);
332
333 // Store the output in the derivative vector
334 for (size_t i = 0; i < dxdt.size(); i++) {
335 dxdt[i] = *(differential_quantity_ptr_pairs[i].second) * (*timestep_ptr);
336 }

352template <typename vector_type>
353struct push_back_state_and_time {
354 private:
355 // Data members
356 vector<vector_type>& states;
357 vector<double>& times;
358 double max_time;
359 double threshold = 0;
360 double threshold_increment = 0.02;
361 string& msg;
362
363 public:
364 // Constructor
365 push_back_state_and_time(
366 vector<vector_type>& states,
367 vector<double>& times,
368 double maximum_time,
369 string& message)
370 : states(states),
371 times(times),
372 max_time(maximum_time),
373 msg(message) {}
374
375 // Operation
376 void operator()(vector_type const& x, double t)
377 {
378 // Add to the message, if required
379 if (t >= max_time || t / max_time >= threshold) {
380 msg += string("Time index = ") +
381 std::to_string(t) +
382 string(" (") +
383 std::to_string(100.0 * t / max_time) +
384 string("%% done) at clock = ") +
385 std::to_string(clock()) +
386 string(" microseconds\n");
387 threshold += threshold_increment;
388 }
389
390 // Store the new values
391 states.push_back(x);
392 times.push_back(t);
393 }

400template <typename vector_type, typename time_type>
401state_vector_map get_results_from_system(
402 std::shared_ptr<dynamical_system> sys,
403 const vector<vector_type>& x_vec,
404 const vector<time_type>& times)
405{
406 state_vector_map results;
407
408 // Get the keys needed for results
409 string_vector output_param_names = sys->get_output_quantity_names();
410
411 // Initialize all result values to all-zero vectors of the correct length
412 vector<double> temporary(x_vec.size());
413 for (const string& p : output_param_names) {
414 results[p] = temporary;
415 }
416
417 // Add an entry for ncalls (hopefully there will be a better way to do this
418 // someday)
419 std::fill(temporary.begin(), temporary.end(), sys->get_ncalls());
420 results["ncalls"] = temporary;
421
422 // Get pointers to the output quantities
423 vector<const double*> output_param_ptrs =
424 sys->get_quantity_access_ptrs(output_param_names);
425
426 // Store the data
427 for (size_t i = 0; i < x_vec.size(); ++i) {
428 sys->update_all_quantities(x_vec[i], times[i]);
429 for (size_t j = 0; j < output_param_names.size(); ++j) {
430 (results[output_param_names[j]])[i] = *(output_param_ptrs[j]);
431 }
432 }
433
434 return results;
435}

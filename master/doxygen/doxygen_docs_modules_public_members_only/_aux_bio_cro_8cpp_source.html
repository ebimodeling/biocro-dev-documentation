14double poisson_density(int x, double lambda)
15{
16 // The probability density for the Poisson distribution is
17 // e^-lambda * lambda^x / x!
18 // The factorial term produces numbers too large to hold, so perform the calculation in the log domain.
19 // n! can be estimated by the approximation sqrt(2 * pi * x) * (x / e)^x.
20 using math_constants::pi;
21 using math_constants::e;
22
23 // Stirling's approximation for n!:
24 double factorial_x = sqrt(2 * pi * x) * pow((x / e), x);
25
26 double log_result = -lambda + x * log(lambda) - log(factorial_x);
27
28 return exp(log_result);
29}

51Light_model lightME(double cosine_zenith_angle, double atmospheric_pressure)
52{
53 double direct_irradiance_transmittance;
54 double diffuse_irradiance_transmittance;
55
56 if (cosine_zenith_angle <= 0) { // Check if the Sun is at or below the horizon. If it is, directly set direct_irradiance_transmittance and diffuse_irradiance_transmittance because the equations below would otherwise produce erroneous results. The Sun is at or below the horizon when cosine_zenith_angle is less than or equal to 0.
57 direct_irradiance_transmittance = 0;
58 diffuse_irradiance_transmittance = 1;
59 } else { // If the Sun is above the horizon, calculate diffuse and direct irradiance from the Sun's angle to the ground and the path through the atmosphere.
60 constexpr double atmospheric_transmittance = 0.85; // dimensionless.
61 const double pressure_ratio = atmospheric_pressure / physical_constants::atmospheric_pressure_at_sea_level; // dimensionless.
62 constexpr double proportion_of_irradiance_scattered = 0.3; // dimensionless.
63
64 direct_irradiance_transmittance = pow(atmospheric_transmittance, (pressure_ratio / cosine_zenith_angle)); // Modified from equation 11.11 of Norman and Campbell.
65 diffuse_irradiance_transmittance = proportion_of_irradiance_scattered * (1 - direct_irradiance_transmittance) * cosine_zenith_angle; // Modified from equation 11.13 of Norman and Campbell.
66 }
67
68 Light_model light_model;
69 light_model.direct_irradiance_fraction = direct_irradiance_transmittance / (direct_irradiance_transmittance + diffuse_irradiance_transmittance); // dimensionless.
70 light_model.diffuse_irradiance_fraction = diffuse_irradiance_transmittance / (direct_irradiance_transmittance + diffuse_irradiance_transmittance); // dimensionless.
71
72 return light_model;
73}

109double absorbed_shortwave_from_incident_ppfd(
110 double incident_ppfd, // micromol / m^2 / s
111 double par_energy_content, // J / micromol
112 double par_energy_fraction, // dimensionless
113 double leaf_reflectance, // dimensionless
114 double leaf_transmittance // dimensionless
115)
116{
117 double incident_par = incident_ppfd * par_energy_content; // J / m^2 / s
118
119 double incident_nir = incident_par *
120 (1 - par_energy_fraction) /
121 par_energy_fraction; // J / m^2 /s
122
123 double incident_shortwave = incident_par + incident_nir; // J / m^2 / s
124
125 return incident_shortwave *
126 (1 - leaf_reflectance - leaf_transmittance) /
127 (1 - leaf_transmittance); // J / m^2 / s
128}

174Light_profile sunML(
175 double ambient_ppfd_beam, // micromol / (m^2 beam) / s
176 double ambient_ppfd_diffuse, // micromol / m^2 / s
177 double lai, // dimensionless from m^2 / m^2
178 int nlayers, // dimensionless
179 double cosine_zenith_angle, // dimensionless
180 double kd, // dimensionless
181 double chil, // dimensionless from m^2 / m^2
182 double absorptivity, // dimensionless from mol / mol
183 double heightf, // m^-1 from m^2 leaf / m^2 ground / m height
184 double par_energy_content, // J / micromol
185 double par_energy_fraction, // dimensionless
186 double leaf_transmittance, // dimensionless
187 double leaf_reflectance // dimensionless
188)
189{
190 if (nlayers < 1 || nlayers > MAXLAY) {
191 throw std::out_of_range("nlayers must be at least 1 but no more than MAXLAY.");
192 }
193 if (cosine_zenith_angle > 1 || cosine_zenith_angle < -1) {
194 throw std::out_of_range("cosine_zenith_angle must be between -1 and 1.");
195 }
196 if (kd > 1 || kd < 0) {
197 throw std::out_of_range("kd must be between 0 and 1.");
198 }
199 if (chil < 0) {
200 throw std::out_of_range("chil must be non-negative.");
201 }
202 if (absorptivity > 1 || absorptivity < 0) {
203 throw std::out_of_range("absorptivity must be between 0 and 1.");
204 }
205 if (heightf <= 0) {
206 throw std::out_of_range("heightf must greater than zero.");
207 }
208
209 // Calculate the leaf shape factor for an ellipsoidal leaf angle
210 // distribution using the equation from page 251 of Campbell & Norman
211 // (1998). We will use this value as `k`, the canopy extinction coefficient
212 // for photosynthetically active radiation throughout the canopy. This
213 // quantity represents the ratio of horizontal area to total area for leaves
214 // in the canopy and is therefore dimensionless from
215 // (m^2 ground) / (m^2 leaf).
216 double zenith_angle = acos(cosine_zenith_angle); // radians
217 double k0 = sqrt(pow(chil, 2) + pow(tan(zenith_angle), 2));
218 double k1 = chil + 1.744 * pow((chil + 1.182), -0.733);
219 double k = k0 / k1; // dimensionless
220
221 double lai_per_layer = lai / nlayers;
222
223 // Calculate the fraction of direct radiation that passes through the canopy
224 // using Equation 15.1. Note that this is equivalent to the fraction of
225 // ground area below the canopy that is exposed to direct sunlight. Note
226 // that if the sun is at or below the horizon, no part of the soil is
227 // sunlit; this corresponds to the case where cosine_zenith_angle is close
228 // to or below zero.
229 double canopy_direct_transmission_fraction =
230 cosine_zenith_angle <= 1E-10 ? 0.0 : exp(-k * lai); // dimensionless
231
232 // Calculate the ambient direct PPFD through a surface parallel to the ground
233 const double ambient_ppfd_beam_ground = ambient_ppfd_beam * cosine_zenith_angle; // micromol / (m^2 ground) / s
234
235 // Calculate the ambient direct PPFD through a unit area of leaf surface
236 double ambient_ppfd_beam_leaf = ambient_ppfd_beam_ground * k; // micromol / (m^2 leaf) / s
237
238 // Start to fill in the light profile values
239 Light_profile light_profile;
240 light_profile.canopy_direct_transmission_fraction = canopy_direct_transmission_fraction;
241
242 // Fill in the layer-dependent light profile values
243 for (int i = 0; i < nlayers; ++i) {
244 // Get the cumulative LAI for this layer, which represents the total
245 // leaf area above this layer
246 const double cumulative_lai = lai_per_layer * (i + 0.5);
247
248 // Calculate the amount of PPFD scattered out of the direct beam using
249 // Equations 15.6 and 15.1 from Campbell & Norman (1998), following
250 // example 15.2. This is a diffuse flux density representing the flux
251 // through any surface.
252 const double scattered_ppfd =
253 ambient_ppfd_beam_ground * (exp(-k * sqrt(absorptivity) * cumulative_lai) -
254 exp(-k * cumulative_lai)); // micromol / m^2 / s
255
256 // Calculate the total flux of diffuse photosynthetically active light
257 // in this layer by combining the scattered PPFD with the ambient
258 // diffuse PPFD. Here we use Equation 15.6 with `alpha` = 1 and
259 // `kbe(phi)` = kd.
260 double diffuse_ppfd =
261 ambient_ppfd_diffuse * exp(-kd * cumulative_lai) + scattered_ppfd; // micromol / m^2 / s
262
263 // Calculate the fraction of sunlit and shaded leaves in this canopy
264 // layer using Equation 15.21.
265 const double Ls = (1 - exp(-k * lai_per_layer)) * exp(-k * cumulative_lai) / k; // dimensionless
266 double sunlit_fraction = Ls / lai_per_layer; // dimensionless
267 double shaded_fraction = 1 - sunlit_fraction; // dimensionless
268
269 // Calculate an "average" incident PPFD for the sunlit and shaded leaves
270 // that doesn't seem to be based on a formula from Campbell & Norman
271 // (1998). It's interpreted as a flux density through a unit of leaf
272 // area, but that may not be correct.
273 double average_ppfd =
274 (sunlit_fraction * (ambient_ppfd_beam_leaf + diffuse_ppfd) + shaded_fraction * diffuse_ppfd) *
275 (1 - exp(-k * lai_per_layer)) / k; // micromol / (m^2 leaf) / s
276
277 // For values of cosine_zenith_angle close to or less than 0, in place
278 // of the calculations above, we want to use the limits of the above
279 // expressions as cosine_zenith_angle approaches 0 from the right:
280 if (cosine_zenith_angle <= 1E-10) {
281 ambient_ppfd_beam_leaf = ambient_ppfd_beam / k1;
282 diffuse_ppfd = ambient_ppfd_diffuse * exp(-kd * cumulative_lai);
283 sunlit_fraction = 0;
284 shaded_fraction = 1;
285 average_ppfd = 0;
286 }
287
288 // Store these values of incident PPFD
289 light_profile.sunlit_incident_ppfd[i] = ambient_ppfd_beam_leaf + diffuse_ppfd; // micromole / (m^2 leaf) / s
290 light_profile.incident_ppfd_scattered[i] = scattered_ppfd; // micromole / m^2 / s
291 light_profile.shaded_incident_ppfd[i] = diffuse_ppfd; // micromole / (m^2 leaf) / s
292 light_profile.average_incident_ppfd[i] = average_ppfd; // micromole / (m^2 leaf) / s
293 light_profile.sunlit_fraction[i] = sunlit_fraction; // dimensionless from m^2 / m^2
294 light_profile.shaded_fraction[i] = shaded_fraction; // dimensionless from m^2 / m^2
295 light_profile.height[i] = (lai - cumulative_lai) / heightf; // m
296
297 // We also need to determine the total amount of absorbed solar energy
298 // for sunlit and shaded leaves
299 light_profile.sunlit_absorbed_shortwave[i] =
300 absorbed_shortwave_from_incident_ppfd(
301 ambient_ppfd_beam_leaf + diffuse_ppfd,
302 par_energy_content,
303 par_energy_fraction,
304 leaf_reflectance,
305 leaf_transmittance); // J / (m^2 leaf) / s
306
307 light_profile.shaded_absorbed_shortwave[i] =
308 absorbed_shortwave_from_incident_ppfd(
309 diffuse_ppfd,
310 par_energy_content,
311 par_energy_fraction,
312 leaf_reflectance,
313 leaf_transmittance); // J / (m^2 leaf) / s
314
315 light_profile.average_absorbed_shortwave[i] =
316 absorbed_shortwave_from_incident_ppfd(
317 average_ppfd,
318 par_energy_content,
319 par_energy_fraction,
320 leaf_reflectance,
321 leaf_transmittance); // J / (m^2 leaf) / s
322 }
323 return light_profile;
324}

440void RHprof(double RH, int nlayers, double* relative_humidity_profile)
441{
442 if (RH > 1 || RH < 0) {
443 throw std::out_of_range("RH must be between 0 and 1.");
444 }

450

642struct ET_Str EvapoTrans2(
643 double absorbed_shortwave_radiation_et, // J / m^2 / s (used to calculate evapotranspiration rate)
644 double absorbed_shortwave_radiation_lt, // J / m^2 / s (used to calculate leaf temperature)
645 double airTemp, // degrees C
646 double RH, // dimensionless from Pa / Pa
647 double WindSpeed, // m / s
648 double stomatal_conductance, // mmol / m^2 / s
649 double leaf_width, // meter
650 double specific_heat_of_air, // J / kg / K
651 double minimum_gbw, // mol / m^2 / s
652 int eteq // unitless parameter
653)
654{
655 const double DdryA = TempToDdryA(airTemp); // kg / m^3. Density of dry air.,
656 const double LHV = TempToLHV(airTemp); // J / kg
657 const double SlopeFS = TempToSFS(airTemp); // kg / m^3 / K
658 const double SWVP = saturation_vapor_pressure(airTemp); // Pa.
659
660 // TODO: This is for about 20 degrees C at 100000 Pa. Change it to use the
661 // model state. (1 * R * temperature) / pressure
662 double constexpr volume_of_one_mole_of_air = 24.39e-3; // m^3 / mol
663
664 double minimum_gbw_in_m_per_s = minimum_gbw * volume_of_one_mole_of_air; // m / s
665
666 if (stomatal_conductance <= 0) {
667 throw std::range_error("Thrown in EvapoTrans2: stomatal conductance is not positive.");
668 }
669
670 double conductance_in_m_per_s = stomatal_conductance * 1e-3 * volume_of_one_mole_of_air; // m / s
671
672 if (RH > 1) {
673 throw std::range_error("Thrown in EvapoTrans2: RH (relative humidity) is greater than 1.");
674 }
675
676 // Convert from vapor pressure to vapor density using the ideal gas law.
677 // This is approximately right for temperatures what won't kill plants.
678 const double SWVC =
679 SWVP / physical_constants::ideal_gas_constant /
680 (airTemp + conversion_constants::celsius_to_kelvin) * physical_constants::molar_mass_of_water; // kg / m^3
681
682 if (SWVC < 0) {
683 throw std::range_error("Thrown in EvapoTrans2: SWVC is less than 0.");
684 }
685
686 const double PsycParam = DdryA * specific_heat_of_air / LHV; // kg / m^3 / K
687
688 const double vapor_density_deficit = SWVC * (1 - RH); // kg / m^3
689
690 const double ActualVaporPressure = RH * SWVP; // Pa
691
692 /* This is the original from WIMOVAC*/
693 double Deltat = 0.01; // degrees C
694 double ga;
695 double rlc; /* Long wave radiation for iterative calculation */
696 {
697 double ChangeInLeafTemp = 10.0; // degrees C
698 double Counter = 0;
699 do {
700 ga = leaf_boundary_layer_conductance_nikolov(
701 WindSpeed, leaf_width, airTemp, Deltat, conductance_in_m_per_s,
702 ActualVaporPressure, minimum_gbw_in_m_per_s); // m / s
703
704 /* In WIMOVAC, ga was added to the canopy conductance */
705 /* ga = (ga * gbcW)/(ga + gbcW); */
706
707 double OldDeltaT = Deltat;
708
709 rlc = 4 * physical_constants::stefan_boltzmann * pow(conversion_constants::celsius_to_kelvin + airTemp, 3) * Deltat; // W / m^2
710
711 /* rlc = net long wave radiation emittted per second
712 * = radiation emitted per second - radiation absorbed per second
713 * = sigma * (Tair + deltaT)^4 - sigma * Tair^4
714 *
715 * To make it a linear function of deltaT, do a Taylor series about
716 * deltaT = 0 and keep only the zero and first order terms.
717 *
718 * rlc = sigma * Tair^4 + deltaT * (4 * sigma * Tair^3) - sigma * Tair^4
719 * = 4 * sigma * Tair^3 * deltaT
720 *
721 * where 4 * sigma * Tair^3 is the derivative of
722 * sigma * (Tair + deltaT)^4 evaluated at deltaT = 0
723 */
724
725 const double PhiN2 = absorbed_shortwave_radiation_lt - rlc; // W / m^2
726
727 /* This equation is from Thornley and Johnson pg. 418 */
728 const double TopValue = PhiN2 * (1 / ga + 1 / conductance_in_m_per_s) - LHV * vapor_density_deficit; // J / m^3
729 const double BottomValue = LHV * (SlopeFS + PsycParam * (1 + ga / conductance_in_m_per_s)); // J / m^3 / K
730 Deltat = fmin(fmax(TopValue / BottomValue, -10), 10); // kelvin. Confine Deltat to the interval [-10, 10]:
731
732 ChangeInLeafTemp = fabs(OldDeltaT - Deltat); // kelvin
733 } while ((++Counter <= 10) && (ChangeInLeafTemp > 0.5));
734 }
735
736 /* Net radiation */
737 const double PhiN = fmax(0, absorbed_shortwave_radiation_et - rlc); // W / m^2
738
739 const double penman_monteith =
740 (SlopeFS * PhiN + LHV * PsycParam * ga * vapor_density_deficit) /
741 (LHV * (SlopeFS + PsycParam * (1 + ga / conductance_in_m_per_s))); // kg / m^2 / s. Thornley and Johnson. 1990. Plant and Crop Modeling. Equation 14.4k. Page 408.
742
743 const double EPen =
744 (SlopeFS * PhiN + LHV * PsycParam * ga * vapor_density_deficit) /
745 (LHV * (SlopeFS + PsycParam)); // kg / m^2 / s
746
747 const double EPries = 1.26 * SlopeFS * PhiN / (LHV * (SlopeFS + PsycParam)); // kg / m^2 / s
748
749 /* Choose equation to report */
750 double TransR;
751 switch (eteq) {
752 case 1:
753 TransR = EPen;
754 break;
755 case 2:
756 TransR = EPries;
757 break;
758 default:
759 TransR = penman_monteith;
760 break;
761 }
762
763 // TransR has units of kg / m^2 / s. Convert to mmol / m^2 / s using the
764 // molar mass of water (in kg / mol) and noting that 1e3 mmol = 1 mol
765 double cf = 1e3 / physical_constants::molar_mass_of_water; // mmol / kg for water
766
767 struct ET_Str et_results;
768 et_results.TransR = TransR * cf; // mmol / m^2 / s
769 et_results.EPenman = EPen * cf; // mmol / m^2 / s
770 et_results.EPriestly = EPries * cf; // mmol / m^2 / s
771 et_results.Deltat = Deltat; // degrees C
772 et_results.boundary_layer_conductance = ga / volume_of_one_mole_of_air; // mol / m^2 / s
773
774 return et_results;
775}

826double leaf_boundary_layer_conductance_nikolov(
827 double windspeed, // m / s
828 double leafwidth, // m
829 double air_temperature, // degrees C
830 double delta_t, // degrees C
831 double stomcond, // m / s
832 double water_vapor_pressure, // Pa
833 double minimum_gbw // m / s
834)
835{
836 constexpr double p = physical_constants::atmospheric_pressure_at_sea_level; // Pa
837
838 double leaftemp = air_temperature + delta_t; // degrees C
839 double gsv = stomcond; // m / s
840 double Tak = air_temperature + conversion_constants::celsius_to_kelvin; // K
841 double Tlk = leaftemp + conversion_constants::celsius_to_kelvin; // K
842 double ea = water_vapor_pressure; // Pa
843 double lw = leafwidth; // m
844
845 double esTl = saturation_vapor_pressure(leaftemp); // Pa.
846
847 // Forced convection
848 constexpr double cf = 1.6361e-3; // TODO: Nikolov et. al equation 29 use cf = 4.322e-3, not cf = 1.6e-3 as is used here.
849
850 double gbv_forced = cf * pow(Tak, 0.56) * pow((Tak + 120) * ((windspeed / lw) / p), 0.5); // m / s.
851
852 // Free convection
853 double gbv_free = gbv_forced;
854 double eb = (gsv * esTl + gbv_free * ea) / (gsv + gbv_free); // Pa. Eq 35
855
856 double Tvdiff = (Tlk / (1 - 0.378 * eb / p)) - (Tak / (1 - 0.378 * ea / p)); // kelvin. It is also degrees C since it is a temperature difference. Eq. 34
857
858 if (Tvdiff < 0) Tvdiff = -Tvdiff;
859
860 gbv_free = cf * pow(Tlk, 0.56) * pow((Tlk + 120) / p, 0.5) * pow(Tvdiff / lw, 0.25); // m / s. Eq. 33
861
862 // Overall conductance
863 double gbv = std::max(gbv_forced, gbv_free); // m / s
864
865 // Apply the minimum
866 return std::max(gbv, minimum_gbw); // m / s
867}

919double leaf_boundary_layer_conductance_thornley(
920 double CanopyHeight, // m
921 double WindSpeed, // m / s
922 double minimum_gbw, // m / s
923 double WindSpeedHeight // m
924)
925{
926 // Define constants used in the model
927 constexpr double kappa = 0.41; // dimensionless. von Karmon's constant. Thornley and Johnson pgs 414 and 416.
928 constexpr double ZetaCoef = 0.026; // dimensionless, Thornley and Johnson 1990, Eq. 14.9o
929 constexpr double ZetaMCoef = 0.13; // dimensionless, Thornley and Johnson 1990, Eq. 14.9o
930 constexpr double dCoef = 0.77; // dimensionless, Thornley and Johnson 1990, Eq. 14.9o.
931 // In the original text this value is reported as 0.64.
932 // In the 2000 reprinting of this text, the authors state that this value should be 0.77.
933 // See "Errata to the 2000 printing" on the page after the preface of the 2000 reprinting of the 1990 text.
934
935 // Apply the height limit
936 CanopyHeight = std::min(CanopyHeight, 0.98 * WindSpeedHeight); // meters
937
938 // Calculate terms that depend on the canopy height
939 const double Zeta = ZetaCoef * CanopyHeight; // meters
940 const double Zetam = ZetaMCoef * CanopyHeight; // meters
941 const double d = dCoef * CanopyHeight; // meters
942
943 // Calculate the boundary layer conductance `ga` according to Thornley and
944 // Johnson Eq. 14.9n, pg. 416
945 const double ga0 = pow(kappa, 2) * WindSpeed; // m / s
946 const double ga1 = log((WindSpeedHeight + Zeta - d) / Zeta); // dimensionless
947 const double ga2 = log((WindSpeedHeight + Zetam - d) / Zetam); // dimensionless
948 const double gbv = ga0 / (ga1 * ga2); // m / s
949
950 // Apply the minimum
951 return std::max(gbv, minimum_gbw); // m / s
952}

954/* Soil Evaporation Function */
955/* Variables I need */
956/* LAI = Leaf Area Index */
957/* k = extinction coefficient */
958/* ppfd = Photosynthetic photon flux density */
959/* awc, wiltp, fieldc = available water content, wilting point and field capacity */
960/* winds = wind speed */
961
962double SoilEvapo(
963 double LAI, double k, double air_temperature, double ppfd,
964 double soil_water_content, double fieldc, double wiltp, double winds,
965 double RelH, double rsec, double soil_clod_size, double soil_reflectance,
966 double soil_transmission, double specific_heat_of_air,
967 double par_energy_content)
968{
969 int method = 1;
970 /* A simple way of calculating the proportion of the soil that is hit by direct radiation. */
971 double soil_area_sunlit_fraction = exp(-k * LAI); // dimensionless.
972
973 /* For now the temperature of the soil will be the same as the air.
974 At a later time this can be made more accurate. I looked at the equations for this and the issue is that it is strongly dependent on
975 depth. Since the soil model now has a single layer, this cannot be implemented correctly at the moment. */
976 double SoilTemp = air_temperature;
977
978 /* From Campbell and Norman. Environmental Biophysics. */
979 /* If relative available water content is */
980 double rawc = (soil_water_content - wiltp) / (fieldc - wiltp); // dimensionless. relative available water content.
981
982 /* Campbell and Norman. Environmental Physics, page 142 */
983 double maximum_uptake_rate = 1 - pow((1 + 1.3 * rawc), -5); // dimensionless
984 /* This is a useful idea because dry soils evaporate little water when dry*/
985
986 /* Total Radiation */
987 /* At the moment soil evaporation is grossly overestimated. In WIMOVAC the light reaching the last layer of leaves is used. Here instead
988 of calculating this again, I will for now assume a 10% as a rough estimate. Note that I could maybe get this since layIdir and
989 layIDiff in sunML are external variables. Rprintf("ppfd %.5f",layIdir[0],"\n"); Update: 03-13-2009. I tried printing this
990 value but it is still too high and will likely overestimate soil evaporation. However, this is still a work in progress.
991 */
992 ppfd *= rsec; /* Radiation soil evaporation coefficient */
993
994 double TotalRadiation = ppfd * par_energy_content;
995
996 double DdryA = TempToDdryA(air_temperature);
997 double LHV = TempToLHV(air_temperature); // J / kg
998 double SlopeFS = TempToSFS(air_temperature);
999 double SWVC = saturation_vapor_pressure(air_temperature) * 1e-5;
1000
1001 double PsycParam = (DdryA * specific_heat_of_air) / LHV;
1002 double vapor_density_deficit = SWVC * (1 - RelH / 100);
1003
1004 double BoundaryLayerThickness = 4e-3 * sqrt(soil_clod_size / winds);
1005 double DiffCoef = 2.126e-5 + 1.48e-7 * SoilTemp;
1006 double SoilBoundaryLayer = DiffCoef / BoundaryLayerThickness;
1007
1008 double Ja = 2 * TotalRadiation * ((1 - soil_reflectance - soil_transmission) / (1 - soil_transmission));
1009
1010 double rlc = 4 * physical_constants::stefan_boltzmann * pow((conversion_constants::celsius_to_kelvin + SoilTemp), 3) * 0.005;
1011 /* the last term should be the difference between air temperature and soil. This is not actually calculated at the moment. Since this is
1012 mostly relevant to the first soil layer where the temperatures are similar. I will leave it like this for now. */
1013
1014 double PhiN = Ja - rlc; /* Calculate the net radiation balance*/
1015 if (PhiN < 0) PhiN = 1e-7;
1016
1017 double Evaporation = 0.0;
1018 if (method == 0) {
1019 /* Priestly-Taylor */
1020 Evaporation = 1.26 * (SlopeFS * PhiN) / (LHV * (SlopeFS + PsycParam)); // kg / m^2 / s.
1021 } else {
1022 /* Penman-Monteith */
1023 Evaporation = (SlopeFS * PhiN + LHV * PsycParam * SoilBoundaryLayer * vapor_density_deficit) / (LHV * (SlopeFS + PsycParam)); // kg / m^2 / s.
1024 }
1025
1026 Evaporation *= soil_area_sunlit_fraction * maximum_uptake_rate; // kg / m^2 / s.
1027 if (Evaporation < 0) Evaporation = 1e-6; // Prevent odd values at very low light levels.
1028
1029 return Evaporation; // kg / m^2 / s.
1030}

1032double compute_wsPhoto(int wsFun, double fieldc, double wiltp, double phi1, double awc) {
1033 // Three different type of equations for modeling the effect of
1034 // water stress on vmax and leaf area expansion. The equation for
1035 // leaf area expansion is more severe than the one for vmax. */
1036 double wsPhoto;
1037 switch (wsFun) {
1038 case 0: { /* linear */
1039 double slp = 1 / (fieldc - wiltp);
1040 double intcpt = 1 - fieldc * slp;
1041 wsPhoto = slp * awc + intcpt;
1042 break;
1043 }
1044 case 1: {
1045 double phi10 = (fieldc + wiltp) / 2;
1046 wsPhoto = 1 / (1 + exp((phi10 - awc) / phi1));
1047 break;
1048 }
1049 case 2: {
1050 double slp = (1 - wiltp) / (fieldc - wiltp);
1051 double intcpt = 1 - fieldc * slp;
1052 double theta = slp * awc + intcpt;
1053 wsPhoto = (1 - exp(-2.5 * (theta - wiltp)/(1 - wiltp))) / (1 - exp(-2.5));
1054 break;
1055 }
1056 case 3:
1057 wsPhoto = 1;
1058 break;
1059 default:
1060 wsPhoto = 0;
1061 }
1062
1063 // wsPhoto can be mathematically lower than zero in some cases but
1064 // I should prevent that:
1065 if (wsPhoto <= 0) {
1066 wsPhoto = 1e-10;
1067 }
1068
1069 // Apparently wsPhoto can be greater than 1.
1070 if (wsPhoto > 1) wsPhoto = 1;
1071
1072 return wsPhoto;
1073}

1085struct ws_str watstr(double precipit, double evapo, double cws, double soildepth, double fieldc,
1086 double wiltp, double soil_saturation_capacity, double sand,
1087 double Ks, double air_entry, double b)
1088{
1089 constexpr double g = 9.8; // m / s^2 ## http://en.wikipedia.org/wiki/Standard_gravity
1090
1091 double precipM = precipit * 1e-3; /* convert precip in mm to m*/
1092 double soil_water_fraction = precipM / soildepth + cws; // m^3 m^-3
1093
1094 double runoff = 0.0;
1095 double Nleach = 0.0; /* Nleach is the NO3 leached. */
1096
1097 if (soil_water_fraction > soil_saturation_capacity) {
1098 runoff = (soil_water_fraction - soil_saturation_capacity) * soildepth; /* This is in meters */
1099 /* Here runoff is interpreted as water content exceeding saturation level */
1100 /* Need to convert to units used in the Parton et al 1988 paper. */
1101 /* The data come in mm/hr and need to be in cm/month */
1102 Nleach = runoff / (1e3 * physical_constants::molar_mass_of_water) * (0.2 + 0.7 * sand);
1103 soil_water_fraction = soil_saturation_capacity;
1104 }
1105
1106 /* The density of water is 0.9982 Mg / m^3 at 20 degrees Celsius. */
1107 /* evapo is demanded water (Mg / ha) */
1108 double npaw = soil_water_fraction - wiltp - evapo / 0.9982 / 1e4 / soildepth; // fraction of saturation.
1109
1110 /* If demand exceeds supply, the crop is getting close to wilting point and transpiration
1111 will be over estimated. In this one layer model though, the crop is practically dead. */
1112 if (npaw < 0) npaw = 0.0;
1113 double awc = npaw + wiltp;
1114
1115 /* Calculating the soil water potential based on equations from Norman and Campbell */
1116 /* tmp.psim = soTexS.air_entry * pow((awc/soTexS.fieldc*1.1),-soTexS.b); */
1117 /* New version of the soil water potential is based on
1118 * "Dynamic Simulation of Water Deficit Effects upon Maize Yield" R. F. Grant Agricultural Systems. 33(1990) 13-39. */
1119 ws_str tmp;
1120 tmp.psim = -exp(log(0.033) + (log(fieldc) - log(awc)) / (log(fieldc) - log(wiltp)) * (log(1.5) - log(0.033))) * 1e3; /* This last term converts from MPa to kPa */
1121
1122 /* This is drainage */
1123 if (awc > fieldc) {
1124 double K_psim = Ks * pow((air_entry / tmp.psim), 2 + 3 / b); /* This is hydraulic conductivity */
1125
1126 double J_w = -K_psim * (-tmp.psim / (soildepth * 0.5)) - g * K_psim; /* Campbell, pg 129 do not ignore the graviational effect. I multiply soil depth by 0.5 to calculate the average depth */
1127 double drainage = J_w * 3600 * 0.9982 * 1e-3; /* This is flow in m^3 / (m^2 * hr). */
1128 awc = awc + drainage / soildepth;
1129 }
1130
1131 tmp.awc = awc;
1132 tmp.runoff = runoff;
1133 tmp.Nleach = Nleach;
1134 return tmp;
1135}

1137/* Function to simulate the multilayer behavior of soil water. In the
1138 future this could be coupled with Campbell (BASIC) ideas to
1139 esitmate water potential. */
1140struct soilML_str soilML(double precipit, double transp, double *cws, double soildepth, double *depths,
1141 double soil_field_capacity, double soil_wilting_point, double soil_saturation_capacity, double soil_air_entry, double soil_saturated_conductivity,
1142 double soil_b_coefficient, double soil_sand_content, double phi1, double phi2, int wsFun,
1143 int layers, double rootDB, double LAI, double k, double AirTemp,
1144 double IRad, double winds, double RelH, int hydrDist, double rfl,
1145 double rsec, double rsdf, double soil_clod_size, double soil_reflectance, double soil_transmission,
1146 double specific_heat_of_air, double par_energy_content)
1147{
1148 constexpr double g = 9.8; /* m / s-2 ## http://en.wikipedia.org/wiki/Standard_gravity */
1149
1150 soilML_str return_value;
1151
1152 /* Crude empirical relationship between root biomass and rooting depth*/
1153 double rootDepth = fmin(rootDB * rsdf, soildepth);
1154
1155 rd_str root_distribution = rootDist(layers, rootDepth, &depths[0], rfl);
1156
1157 /* unit conversion for precip */
1158 double oldWaterIn = 0.0;
1159 double waterIn = precipit * 1e-3; /* convert precip in mm to m */
1160
1161 double drainage = 0.0;
1162 double wsPhotoCol = 0.0;
1163 double LeafWSCol = 0.0;
1164 double Sevap = 0.0;
1165 double oldEvapoTra = 0.0;
1166
1167 for (int i = layers - 1; i >= 0; --i) { /* The counter, i, decreases because I increase the water content due to precipitation in the last layer first*/
1168
1169 double layerDepth = depths[i+1] - depths[i]; /* This supports unequal depths. */
1170
1171 if (hydrDist > 0) {
1172 /* For this section see Campbell and Norman "Environmental BioPhysics" Chapter 9*/
1173 /* First compute the matric potential */
1174 double psim1 = soil_air_entry * pow((cws[i]/soil_saturation_capacity), -soil_b_coefficient); /* This is matric potential of current layer */
1175 double dPsim;
1176 if (i > 0) {
1177 double psim2 = soil_air_entry * pow((cws[i-1]/soil_saturation_capacity), -soil_b_coefficient); /* This is matric potential of next layer */
1178 dPsim = psim1 - psim2;
1179 /* The substraction is from the layer i - (i-1). If this last term is positive then it will move upwards. If it is negative it will move downwards. Presumably this term is almost always positive. */
1180 } else {
1181 dPsim = 0.0;
1182 }
1183
1184 double K_psim = soil_saturated_conductivity * pow((soil_air_entry/psim1), 2+3/soil_b_coefficient); /* This is hydraulic conductivity */
1185 double J_w = K_psim * (dPsim/layerDepth) - g * K_psim; /* Campbell, pg 129 do not ignore the graviational effect*/
1186 /* Notice that K_psim is positive because my reference system is reversed */
1187 /* This last result should be in kg/(m2 * s) */
1188 J_w *= 3600 * 0.9882 * 1e-3; /* This is flow in m3 / (m^2 * hr). */
1189
1190 if (i == (layers-1) && J_w < 0) {
1191 /* cws[i] = cws[i] + J_w / layerDepth; Although this should be done it drains the last layer too much.*/
1192 drainage += J_w;
1193 } else {
1194 cws[i] = cws[i] - J_w / layerDepth;
1195 if (i > 0) {
1196 cws[i - 1] = cws[i - 1] + J_w / layerDepth;
1197 }
1198 }
1199 }
1200
1201 if (cws[i] > soil_saturation_capacity) cws[i] = soil_saturation_capacity;
1202 if (cws[i] < soil_wilting_point) cws[i] = soil_wilting_point;
1203 // Here is a convention aw is available water in volume and awc
1204 // is available water content as a fraction of the soil section being investigated.
1205 double aw = cws[i] * layerDepth;
1206 /* Available water (for this layer) is the current water status times the layer depth */
1207
1208 if (waterIn > 0) {
1209 /* There is some rain. Need to add it.*/
1210 aw += waterIn / layers + oldWaterIn; /* They are both in meters so it works */
1211 /* Adding the same amount to water to each layer */
1212 /* In case there is overflow */
1213 double diffw = soil_field_capacity * layerDepth - aw;
1214
1215 if (diffw < 0) {
1216 /* This means that precipitation exceeded the capacity of the first layer */
1217 /* Save this amount of water for the next layer */
1218 oldWaterIn = -diffw;
1219 aw = soil_field_capacity * layerDepth;
1220 } else {
1221 oldWaterIn = 0.0;
1222 }
1223 }
1224
1225 /* Root Biomass */
1226 double rootATdepth = rootDB * root_distribution.rootDist[i];
1227 return_value.rootDist[i] = rootATdepth;
1228 /* Plant available water is only between current water status and permanent wilting point */
1229 /* Plant available water */
1230 double pawha = (aw - soil_wilting_point * layerDepth) * 1e4;
1231
1232 if (pawha < 0) pawha = 0;
1233
1234 double Ctransp = 0.0;
1235 double EvapoTra = 0.0;
1236 double Newpawha;
1237
1238 if (i == 0) {
1239 /* Only the first layer is affected by soil evaporation */
1240 double awc2 = aw / layerDepth;
1241 /* SoilEvapo function needs soil water content */
1242 Sevap = SoilEvapo(
1243 LAI, k, AirTemp, IRad, awc2, soil_field_capacity,
1244 soil_wilting_point, winds, RelH, rsec, soil_clod_size,
1245 soil_reflectance, soil_transmission, specific_heat_of_air,
1246 par_energy_content) * 3600 * 1e-3 * 10000; // Mg / ha / hr. 3600 s / hr * 1e-3 Mg / kg * 10000 m^2 / ha.
1247 /* I assume that crop transpiration is distributed simlarly to
1248 root density. In other words the crop takes up water proportionally
1249 to the amount of root in each respective layer.*/
1250 Ctransp = transp * root_distribution.rootDist[0];
1251 EvapoTra = Ctransp + Sevap;
1252 constexpr double density_of_water_at_20_celcius = 0.9982; // Mg m^-3.
1253 Newpawha = pawha - EvapoTra / density_of_water_at_20_celcius;
1254 /* The first term in the rhs pawha is the m3 of water available in this layer.
1255 EvapoTra is the Mg H2O ha-1 of transpired and evaporated water. 1/0.9882 converts from Mg to m3 */
1256 } else {
1257 Ctransp = transp * root_distribution.rootDist[i];
1258 EvapoTra = Ctransp;
1259 Newpawha = pawha - (EvapoTra + oldEvapoTra);
1260 }
1261
1262 if (Newpawha < 0) {
1263 /* If the Demand is not satisfied by this layer. This will be stored and added to subsequent layers*/
1264 oldEvapoTra = -Newpawha;
1265 aw = soil_wilting_point * layerDepth;
1266 }
1267
1268 double awc = Newpawha / 1e4 / layerDepth + soil_wilting_point;
1269
1270 /* This might look like a weird place to populate the structure, but is more convenient*/
1271 return_value.cws[i] = awc;
1272
1273 // To-do: Replace this block with a call to compute_wsPhoto.
1274 /* three different type of equations for modeling the effect of water stress on vmax and leaf area expansion.
1275 The equation for leaf area expansion is more severe than the one for vmax. */
1276 double wsPhoto = 0.0;
1277 double slp = 0.0;
1278 double intcpt = 0.0;
1279 double theta = 0.0;
1280
1281 if (wsFun == 0) { /* linear */
1282 slp = 1/(soil_field_capacity - soil_wilting_point);
1283 intcpt = 1 - soil_field_capacity * slp;
1284 wsPhoto = slp * awc + intcpt;
1285 } else if (wsFun == 1) {
1286 double phi10 = (soil_field_capacity + soil_wilting_point)/2;
1287 wsPhoto = 1/(1 + exp((phi10 - awc)/ phi1));
1288 } else if (wsFun == 2) {
1289 slp = (1 - soil_wilting_point)/(soil_field_capacity - soil_wilting_point);
1290 intcpt = 1 - soil_field_capacity * slp;
1291 theta = slp * awc + intcpt;
1292 wsPhoto = (1 - exp(-2.5 * (theta - soil_wilting_point)/(1 - soil_wilting_point))) / (1 - exp(-2.5));
1293 } else if (wsFun == 3) {
1294 wsPhoto = 1;
1295 }
1296
1297 if (wsPhoto <= 0 )
1298 wsPhoto = 1e-10; /* This can be mathematically lower than zero in some cases but I should prevent that. */
1299
1300 wsPhotoCol += wsPhoto;
1301
1302 double LeafWS = pow(awc, phi2) * 1/pow(soil_field_capacity, phi2);
1303 if (wsFun == 3) {
1304 LeafWS = 1;
1305 }
1306 LeafWSCol += LeafWS;
1307
1308 }
1309
1310 if (waterIn > 0) {
1311 drainage = waterIn;
1312 /* Need to convert to units used in the Parton et al 1988 paper. */
1313 /* The data comes in mm/hr and it needs to be in cm/month */
1314 return_value.Nleach = drainage * 0.1 * (1/24 * 30) / (1e3 * physical_constants::molar_mass_of_water * (0.2 + 0.7 * soil_sand_content));
1315 }
1316 else {
1317 return_value.Nleach = 0.0;
1318 }
1319
1320
1321 return_value.rcoefPhoto = (wsPhotoCol/layers);
1322 return_value.drainage = drainage;
1323 return_value.rcoefSpleaf = (LeafWSCol/layers);
1324 return_value.SoilEvapo = Sevap;
1325
1326 return return_value;
1327}

1446struct rd_str rootDist(int n_layers, double rootDepth, double *depths, double rfl)
1447{
1448 /*
1449 * Calculate the fraction of total root mass for each layer in `depths` assuming the mass
1450 * is follows a Poisson distribution along the depth.
1451 *
1452 * Returns a struct with one array of size `n_layers`.
1453 * Each element in the array is the fraction of total root mass in that layer.
1454 * The sum of all elements of the result equals 1.
1455 */
1456
1457 double layerDepth = 0.0;
1458 double CumLayerDepth = 0.0;
1459 int CumRootDist = 1;
1460 double rootDist[n_layers];
1461 double cumulative_a = 0.0;
1462
1463 for (int i = 0; i < n_layers; ++i) {
1464 if (i == 0) {
1465 layerDepth = depths[1];
1466 } else {
1467 layerDepth = depths[i] - depths[i-1];
1468 }
1469
1470 CumLayerDepth += layerDepth;
1471
1472 if (rootDepth > CumLayerDepth) {
1473 CumRootDist++;
1474 }
1475 }
1476
1477 for (int j = 0; j < n_layers; ++j) {
1478 if (j < CumRootDist) {
1479 double a = poisson_density(j + 1, (double)CumRootDist * rfl);
1480 rootDist[j] = a;
1481 cumulative_a += a;
1482 } else {
1483 rootDist[j] = 0;
1484 }
1485 }
1486
1487 struct rd_str result;
1488 for (int k = 0; k < n_layers; ++k) {
1489 result.rootDist[k] = rootDist[k] / cumulative_a;
1490 }
1491 return result;
1492}

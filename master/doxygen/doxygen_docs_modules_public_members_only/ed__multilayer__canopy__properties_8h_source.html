210void ed_multilayer_canopy_properties::run() const
211{
212 // Calculate PAR levels throughout the canopy. Note that sunML specifies that its
213 // `Idir` and `Idiff` arguments should have [micromol / m^2 / s] units. However,
214 // we can leave these quantities as energy flux densities with [J / m^2 / s] units
215 // because the two units are simply related by a multiplicative conversion factor
216 // and the output of sunML is linear with respect to `Idir` and `Idiff`.
217 struct Light_profile par_profile = sunML(
218 par_incident_direct, par_incident_diffuse, lai, nlayers,
219 cosine_zenith_angle, kd, chil, absorptivity_par, heightf,
220 par_energy_content, par_energy_fraction, leaf_transmittance,
221 leaf_reflectance);
222
223 // Calculate relative humidity levels throughout the canopy
224 double relative_humidity_profile[nlayers];
225 RHprof(rh, nlayers, relative_humidity_profile); // Modifies relative_humidity_profile.
226
227 // Calculate windspeed throughout the canopy
228 double wind_speed_profile[nlayers];
229 WINDprof(windspeed, lai, nlayers, wind_speed_profile); // Modifies wind_speed_profile.
230
231 // Calculate leaf nitrogen throughout the canopy
232 double leafN_profile[nlayers];
233 LNprof(LeafN, lai, nlayers, kpLN, leafN_profile); // Modifies leafN_profile.
234
235 // Update layer-dependent outputs
236 //
237 // par_profile.sunlit_incident_ppfd and par_profile.shaded_incident_ppfd represent
238 // incident PAR energy flux densities in units of J / m^2 / s. However, the
239 // Collatz model expects a quantum flux in units of mol / m^2 / s. So we use
240 // the `par_energy_content` quantity to convert from J to micromol, and then
241 // an additional factor of 1e6 micromol / mol is required.
242 //
243 // par_profile.average_incident_ppfd represents some kind of wonky "average" incident
244 // PAR flux on the leaves, but the evapotranspiration code expects a value for
245 // total absorbed solar energy (including NIR). To convert, we use the method from
246 // the Evapotrans2 code. First we use the `par_energy_fraction` quantity
247 // to determine the total amount of solar energy incident on the leaf (including NIR).
248 // Then we determine the amount of light absorbed by the leaf using the following factor:
249 // (1 - LeafReflectance - tau) / (1 - tau)
250 double const absorb_factor = (1 - leaf_reflectance - leaf_transmittance) / (1 - leaf_transmittance);
251 for (int i = 0; i < nlayers; ++i) {
252 update(sunlit_fraction_ops[i], par_profile.sunlit_fraction[i]);
253 update(shaded_fraction_ops[i], par_profile.shaded_fraction[i]);
254 update(height_ops[i], par_profile.height[i]);
255 update(sunlit_incident_par_ops[i], par_profile.sunlit_incident_ppfd[i]);
256 update(sunlit_collatz_PAR_flux_ops[i], par_profile.sunlit_incident_ppfd[i] / par_energy_content * 1e-6);
257 update(incident_scattered_par_ops[i], par_profile.incident_ppfd_scattered[i]);
258 update(shaded_incident_par_ops[i], par_profile.shaded_incident_ppfd[i]);
259 update(shaded_collatz_PAR_flux_ops[i], par_profile.shaded_incident_ppfd[i] / par_energy_content * 1e-6);
260 update(incident_average_par_ops[i], par_profile.average_incident_ppfd[i]);
261 update(solar_energy_absorbed_leaf_ops[i], par_profile.average_incident_ppfd[i] / par_energy_fraction * absorb_factor);
262 update(rh_ops[i], relative_humidity_profile[i]);
263 update(windspeed_ops[i], wind_speed_profile[i]);
264 update(LeafN_ops[i], leafN_profile[i]);
265 }
266}
